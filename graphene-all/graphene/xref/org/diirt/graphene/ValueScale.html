5   package org.diirt.graphene;
6   
7   import org.diirt.util.stats.Range;
8   
9   /**
10   * The scale to be used to create axis references and rescale values.
11   *
12   * @author carcassi
13   */
14  public interface ValueScale {
15      
16      /**
17       * Scales the actual value to the scale value.
18       * 
19       * @param value actual value to be scaled
20       * @param minValue actual range min
21       * @param maxValue actual range max
22       * @param newMinValue scale range min
23       * @param newMaxValue scale range max
24       * @return new value in the scale
25       */
26      double scaleValue(double value, double minValue, double maxValue, double newMinValue, double newMaxValue);
27      
28      /**
29       * Scales the scale value to the actual value.
30       * 
31       * @param scaleValue scale value to be scaled
32       * @param minValue actual range min
33       * @param maxValue actual range max
34       * @param newMinValue scale range min
35       * @param newMaxValue scale range max
36       * @return new value in the actual range
37       */
38      double invScaleValue(double scaleValue, double minValue, double maxValue, double newMinValue, double newMaxValue);
39      
40      /**
41       * Returns the reference axes for a given range
42       * @param range the range of values to create a reference axis for
43       * @param minRefs minimum references
44       * @param maxRegs max references
45       * @return reference axes
46       */
47      ValueAxis references(Range range, int minRefs, int maxRegs);
48  }
