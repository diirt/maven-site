5   package org.diirt.graphene;
6   
7   import org.diirt.util.stats.Statistics;
8   import java.util.List;
9   import org.diirt.util.array.ListNumber;
10  import org.diirt.util.time.TimeInterval;
11  import org.diirt.util.time.Timestamp;
12  
13  /**
14   *
15   * @author carcassi
16   */
17  public interface TimeSeriesDataset {
18      
19      /**
20       * The values.
21       * <p>
22       * If the dataset is empty, it returns an empty list.
23       * 
24       * @return the values; never null
25       */
26      public ListNumber getValues();
27      
28      /**
29       * The time for the values.
30       * <p>
31       * If the dataset is empty, it returns an empty list.
32       * 
33       * @return time; never null
34       */
35      public List<Timestamp> getTimestamps();
36  
37      /**
38       * Returns the time normalized within the range of the dataset.
39       * <p>
40       * If the dataset is empty, it returns an empty list.
41       * 
42       * @param normalizationRange the interval over which the data is going to be displayed
43       * @return normalized time; never null
44       */
45      public ListNumber getNormalizedTime( TimeInterval normalizationRange );
46      
47      /**
48       * The statistical information of the values.
49       * <p>
50       * If the dataset is empty, or if it contains only NaN values, it returns null.
51       * 
52       * @return statistical information; null if no actual values in the dataset
53       */
54      public Statistics getStatistics();
55      
56      /**
57       * The interval of time where the data is defined.
58       * <p>
59       * If the dataset is empty, or if it contains only NaN values, it returns null.
60       * 
61       * @return x statistical information; null if no actual values in the dataset
62       */
63      public TimeInterval getTimeInterval();
64      
65      /**
66       * The number of values in the dataset.
67       * <p>
68       * This number matches the size of the list returned by {@link #getValues() }.
69       * 
70       * @return the number of values in this dataset
71       */
72      public int getCount();
73  }
