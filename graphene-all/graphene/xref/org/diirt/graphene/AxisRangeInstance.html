5   package org.diirt.graphene;
6   
7   import org.diirt.util.stats.Range;
8   
9   /**
10   * A stateful object that computes the value range for a particular graph.
11   * It includes the state to compute the range for a particular axis on a particular
12   * graph.
13   * <p>
14   * Objects of this type are mutable, and cannot be shared
15   * across different graphs, on different threads.
16   *
17   * @author carcassi
18   */
19  public interface AxisRangeInstance {
20      
21      /**
22       * Calculates the range for the axis, given the range of the data and
23       * the suggested range.
24       * 
25       * @param dataRange the actual range of the data being displayed
26       * @param displayRange the suggested range for the data
27       * @return the range to use on the axis
28       */
29      public Range axisRange(Range dataRange, Range displayRange);
30      
31      /**
32       * The AxisRange of which this object is an instance of.
33       * 
34       * @return an AxisRange; never null
35       */
36      public AxisRange getAxisRange();
37  }
