5   package org.diirt.graphene;
6   
7   /**
8    * An instance of a {@link NumberColorMap} that associates a numeric value
9    * to an RGB color.
10   *
11   * @author sjdallst
12   */
13  public interface NumberColorMapInstance {
14      
15      /**
16       * Returns the color associated to the value. The color is an integer
17       * enconding an ARGB (Alpha-Red-Green-Blue) value. Each component is 8 bits,
18       * making 32 bit total.
19       * 
20       * @param value the value to be mapped
21       * @return the color corresponding to the value
22       */
23      public int colorFor(double value);
24  }
