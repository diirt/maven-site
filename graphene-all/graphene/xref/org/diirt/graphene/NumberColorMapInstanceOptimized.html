7   import org.diirt.util.stats.Range;

9   /**
10   * An optimized instance of a color map, where colors are pre-calculated.
11   * <p>
12   * TODO: allow choice of number of colors
13   * 
14   * @author sjdallst
15   */
16  class NumberColorMapInstanceOptimized implements NumberColorMapInstance {
17  
18      private int arrayLength = 1000;
19      private int[] colors = new int[arrayLength];
20      private int nanColor;
21      private Range range;
22      private double max, min, total;
23      private final String name;
24  
25      NumberColorMapInstanceOptimized(NumberColorMapInstance instance, Range range) {
26          min = range.getMinimum();
27          max = range.getMaximum();
28          total = max - min;
29          for (int i = 0; i < arrayLength; i++) {
30              //account for possible rounding errors on last entry.
31              if (i == arrayLength - 1) {
32                  colors[i] = instance.colorFor(max);
33              } else {
34                  colors[i] = instance.colorFor(min + i * (total / ((double) (arrayLength - 1))));
35              }
36          }
37          this.range = range;
38          this.name = instance.toString() + " opt(" + arrayLength + ")";
39      }
40  
41      // TODO: what is this doing?
42      NumberColorMapInstanceOptimized(NumberColorMapInstance instance, Range oldRange, Range newRange) {
43          double oldMin = oldRange.getMinimum();
44          double oldMax = oldRange.getMaximum();
45          double oldTotal = oldMax - oldMin;
46          for (int i = 0; i < arrayLength; i++) {
47              //account for possible rounding errors on last entry.
48              if (i == arrayLength - 1) {
49                  colors[i] = instance.colorFor(oldMax);
50              } else {
51                  colors[i] = instance.colorFor(oldMin + i * (oldTotal / ((double) (arrayLength - 1))));
52              }
53          }
54          min = newRange.getMinimum();
55          max = newRange.getMaximum();
56          total = max - min;
57          this.range = newRange;
58          this.name = instance.toString() + " opt2(" + arrayLength + ")";
59      }
60  
61      @Override
62      public int colorFor(double value) {
63          int index = (int) ((value - min) / total * (arrayLength - 1));
64          if (index < 0) {
65              index = 0;
66          } else if (index >= colors.length) {
67              index = colors.length - 1;
68          }
69          return colors[index];
70      }
71  
72      @Override
73      public String toString() {
74          return name;
75      }
76      
77  }
