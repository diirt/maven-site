5   package org.diirt.graphene;
6   
7   /**
8    * A set of parameters that can be applied to a <code>BubbleGraph2DRendererUpdate</code>
9    * to update its settings.
10   * 
11   * <p>
12   * Only the parameters that are set in the update get applied to the renderer.
13   * Parameters unique to the bubble graph that can be changed are:
14   * <ul>
15   *   <li>focusPixel - TODO</li>
16   *   <li>highlightFocusValue - TODO</li>
17   * </ul>
18   * 
19   * @author carcassi
20   */
21  public class BubbleGraph2DRendererUpdate extends Graph2DRendererUpdate<BubbleGraph2DRendererUpdate> {
22      
23      private Integer focusPixelX;
24      private Integer focusPixelY;
25      private Boolean highlightFocusValue;
26      
27      /**
28       * Changes the current position of the pixel with focus.
29       * 
30       * @param x horizontal position of the pixel
31       * @param y vertical position of the pixel
32       * @return this
33       */
34      public BubbleGraph2DRendererUpdate focusPixel(int x, int y) {
35          this.focusPixelX = x;
36          this.focusPixelY = y;
37          return this;
38      }
39      
40      /**
41       * Sets the state of highlightFocusValue.
42       * <ul>
43       *  <li>True - highlight and show the value the mouse is on.</li>
44       *  <li>False - Avoid calculation involved with finding the highlighted value/ do not highlight the value.</li>
45       * </ul>
46       * 
47       * @param highlightFocusValue true if value should be highlighted
48       * @return this
49       */
50      public BubbleGraph2DRendererUpdate highlightFocusValue(boolean highlightFocusValue) {
51          this.highlightFocusValue = highlightFocusValue;
52          return this;
53      }
54      
55      /**
56       * New horizontal position of the focused pixel.
57       * 
58       * @return new horizontal position or null
59       */
60      public Integer getFocusPixelX() {
61          return focusPixelX;
62      }
63      
64      /**
65       * New vertical position of the focused pixel.
66       * 
67       * @return new vertical position or null
68       */
69      public Integer getFocusPixelY() {
70          return focusPixelY;
71      }
72  
73      /**
74       * Whether to highlight the focused value.
75       * 
76       * @return the new setting or null
77       */
78      public Boolean getHighlightFocusValue() {
79          return highlightFocusValue;
80      }
81  }
