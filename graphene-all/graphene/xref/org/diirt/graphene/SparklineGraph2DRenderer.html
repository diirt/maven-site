13  /**
14   * Creates a renderer that is capable of drawing a sparkline graph.
15   * 
16   * <p>
17   * A sparkline graph
18   * <ul>
19   *      <li>Displays a line</li>
20   *      <li>Does not have axes labels</li>
21   *      <li>Does not have scales display on the axes</li>
22   *      <li>Often draws small circles at important values on the line</li>
23   *      <li>Important values are:
24   *          <ul>
25   *              <li>First Value</li>
26   *              <li>Last Value</li>
27   *              <li>Maximum Value</li>
28   *              <li>Minimum Value</li>
29   *          </ul>
30   *      </li>
31   *      <li>Often maintains a high width to height aspect ratio.
32   *          This can be manually setting the appropriate image width and height,
33   *          or can be applied in the code through the <code>aspectRatio</code>.
34   *          The aspect ratio should be set so that the line of the sparkline
35   *          never has a slope greater than 45-degrees.</li>
36   * </ul>
37   * 
38   * @author asbarber
39   * @author jkfeng
40   * @author sjdallst
41   */
42  public class SparklineGraph2DRenderer extends Graph2DRenderer<SparklineGraph2DRendererUpdate>{

44      /**
45       * Creates a new sparkline graph renderer.
46       * Will draw a circle at the max value, min value, and last value.
47       * 
48       * @param imageWidth the graph width in pixels
49       * @param imageHeight the graph height in pixels
50       */    
51      public SparklineGraph2DRenderer(int imageWidth, int imageHeight){
52          super(imageWidth, imageHeight);
53          super.xLabelMargin = 0;
54          super.yLabelMargin = 0; 
55          
56          super.rightMargin = 0;
57          super.leftMargin = 0;
58          super.bottomMargin = 0;
59          super.topMargin = 0;
60          
61          //Set all area matgins to 1 to account for the drawing of circles.
62          super.leftAreaMargin   = 1;
63          super.rightAreaMargin  = 1;
64          super.bottomAreaMargin = 1;
65          super.topAreaMargin    = 1;        
66      }

70      private Color   minValueColor = new Color(28, 160, 232),    //Blue
71                      maxValueColor = new Color(28, 160, 232),    
72                      firstValueColor = new Color(223, 59, 73),   //Red         
73                      lastValueColor = new Color(223, 59, 73);
74      private boolean drawCircles = true;
75      
76      //Min, Max, Last Values and Indices
77      private int     maxIndex, 
78                      minIndex,
79                      firstIndex,
80                      lastIndex;
81      private double  maxValueY = -1, 
82                      minValueY = -1,
83                      firstValueY = -1,
84                      lastValueY = -1;
85      private Double  aspectRatio = null;

99      /**
100      * Draws the graph on the given graphics context.
101      * The render process is:
102      * <ul>
103      *      <li>Set aspect ratio</li>
104      *      <li>Calculate data and plot ranges</li>
105      *      <li>Calculates the region for the graph</li>
106      *      <li>Draws the background</li>
107      *      <li>Draws the line</li>
108      *      <li>Draw circles if necessary</li>
109      * </ul>
110      * 
111      * A circle is drawn at the max value and min value.
112      * A circle is drawn at the first value and last value.
113      * Each circle is drawn at 70% transparency.
114      * If two circles are set to draw at the same value, only one circle is drawn.
115      * If there is overlap, first/last values are drawn instead of max/min values.
116      * 
117      * @param g the graphics on which to display the data
118      * @param data the data to display
119      */
120     public void draw(Graphics2D g, Point2DDataset data) {
121         this.g = g;
122         
123         /*If we want to use the aspect ratio, we change the start and end of the coordinate plot,
124         so that the total height is equal to the width of the xplot divided by the aspect ratio.*/  
125         if(aspectRatio != null){
126             adjustGraphToAspectRatio();
127         }
128         
129         //General Rendering
130         calculateRanges(data.getXStatistics().getRange(), data.getXDisplayRange(), data.getYStatistics().getRange(), data.getYDisplayRange());
131         calculateGraphArea();
132 
133         drawBackground();
134         g.setColor(Color.BLACK);        
135   
136         //Calculates data values
137         SortedListView xValues = org.diirt.util.array.ListNumbers.sortedView(data.getXValues());
138         ListNumber yValues = org.diirt.util.array.ListNumbers.sortedView(data.getYValues(), xValues.getIndexes());        
139         setClip(g);
140         
141         //Draws Line  
142         g.setRenderingHint(RenderingHints.KEY_STROKE_CONTROL, RenderingHints.VALUE_STROKE_DEFAULT);
143         g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);        
144         drawValueExplicitLine(xValues, yValues, interpolation, ReductionScheme.FIRST_MAX_MIN_LAST);
145         
146         //Draws a circle at the max, min, and last value
147         if(drawCircles){
148             //Hints: pure stroke, no antialias
149             g.setRenderingHint(RenderingHints.KEY_STROKE_CONTROL, RenderingHints.VALUE_STROKE_PURE);
150             g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF);
151             
152             //Set transparency
153             AlphaComposite ac = java.awt.AlphaComposite.getInstance(AlphaComposite.SRC_OVER, 0.7F);
154             g.setComposite(ac);        
155             
156             //Fills circle
157             if (!hasOverlapMinimum()){
158                 drawCircle(g, data, minIndex, minValueColor);
159             }
160             if (!hasOverlapMaximum()){
161                 drawCircle(g, data, maxIndex, maxValueColor);
162             }
163             drawCircle(g, data, firstIndex, firstValueColor);
164             drawCircle(g, data, lastIndex, lastValueColor);
165         }
166     }
167     
168     /**
169      * Determines whether the minimum value circle overlaps with the first or last values.
170      * This is useful in determining whether to not draw the minimum circle.
171      * 
172      * @return whether the "minimum" circle overlaps with other circles
173      */
174     private boolean hasOverlapMinimum(){
175         return minIndex == lastIndex || minIndex == firstIndex;
176     }
177     
178     /**
179      * Determines whether the maximum value circle overlaps with the first or last values.
180      * This is useful in determining whether to not draw the maximum circle.
181      * 
182      * @return whether the "maximum" circle overlaps with other circles
183      */    
184     private boolean hasOverlapMaximum(){
185         return maxIndex == lastIndex || maxIndex == firstIndex;
186     }
187 
188     /**
189      * Draws a circle of a certain color at a certain position.
190      * The pixel position is determined from the data value.
191      * The data value is found by the 'index' element in 'data'.
192      * 
193      * @param g graphic context to draw (where circles are drawn)
194      * @param data set of data that is graphed
195      * @param index position in the data set where the circle is drawn (index position)
196      * @param color color of circle that is drawn
197      */
198     protected void drawCircle(Graphics2D g, Point2DDataset data, int index, Color color){
199             double x = Math.floor(scaledX(data.getXValues().getDouble(index)))+.5;
200             double y = Math.floor(scaledY(data.getYValues().getDouble(index)))+.5;
201             g.setColor(color);
202             Shape circle = createShape(x, y, circleDiameter);
203             g.fill(circle);        
204     }
205     
206     /**
207      * Creates a circle shape at the given position with given size.
208      * @param x x position of shape
209      * @param y y position of shape
210      * @param size diameter of circle
211      * @return ellipse (circle) shape
212      */
213     protected Shape createShape(double x, double y, double size) {
214         double halfSize = size / 2;
215         Ellipse2D.Double circle = new Ellipse2D.Double(x-halfSize, y-halfSize, size, size);
216         return circle;
217     }

219     /**
220      * Operations for every value in data as it is processed.
221      * This checks whether each new value is an important index.
222      * An important index is if this y-value is a maximum, minimum, first, or last.
223      * Note: this y-value is always the last value.
224      * 
225      * @param index element position in data set
226      * @param valueX x value being processed
227      * @param valueY y value being processed
228      * @param scaledX x value scaled by data
229      * @param scaledY y value scaled by data
230      */
231     @Override
232     protected void processScaledValue(int index, double valueX, double valueY, double scaledX, double scaledY) {
233         //Checks if new value is the new min or the new max
234         
235         //Base Case
236         if (index == 0){
237             firstIndex = 0;
238             firstValueY = valueY;
239             
240             maxValueY = valueY;
241             minValueY = valueY;
242         }
243         else{
244             //Max
245             if (maxValueY <= valueY){
246                 maxValueY = valueY;
247                 maxIndex = index;
248             }
249             //Min
250             if (minValueY >= valueY){
251                 minValueY = valueY;
252                 minIndex = index;
253             }  
254         }
255         
256         //New point is always last point
257         lastValueY = valueY;
258         lastIndex = index;
259     }

261     /**
262      * Applies the update to the renderer.
263      * 
264      * @param update the update to apply
265      */    
266     @Override
267     public void update(SparklineGraph2DRendererUpdate update) {
268         super.update(update);
269 
270         //Applies updates to members of this class
271         if (update.getMinValueColor() != null){
272             minValueColor = update.getMinValueColor();
273         }
274         if (update.getMaxValueColor() != null){
275             maxValueColor = update.getMaxValueColor();
276         }
277         if (update.getCircleDiameter() != null){
278             circleDiameter = update.getCircleDiameter();
279         }
280         if (update.getDrawCircles() != null){
281             drawCircles = update.getDrawCircles();
282         }
283         if (update.getInterpolation() != null) {
284             interpolation = update.getInterpolation();
285         } 
286         if (update.getAspectRatio() != null){
287             aspectRatio = update.getAspectRatio();
288         }
289     }

428     /**
429      * Adjusts the area margins to maintain the aspect ratio.
430      * The aspect ratio is applied only to the graph area (margins are ignored).
431      * 
432      * The area margin for all borders is set to 1 by default.
433      * 
434      * The left/right area margins are increased if the width needs to shrink to maintain the ratio.
435      * The top/bottom area margins are increased if the height needs to shrink to maintain the ratio.
436      * Note that the width and height can never be increased, so the aspect ratio is maintained by shrinking axes.
437      * 
438      * <p>
439      * Example:
440      *  Width = 100 pixels
441      *  Height = 20 pixels
442      *  Ratio is 4 : 1  (W : H)
443      * 
444      * The width and height could then be 100 : 25 or 80 : 20
445      * Since the height cannot be increased from 20 to 25, the first option would not work.
446      * The option 80 : 20 is then set.
447      * The width is decreased from 100 to 80 by setting the left and right area margin to 10 pixels.
448      * </p>
449      */
450     private void adjustGraphToAspectRatio(){
451         //Only looks at available graph area region
452         int relevantHeight = super.getImageHeight() - bottomMargin - topMargin,
453             relevantWidth  = super.getImageWidth() - rightMargin - leftMargin;
454         
455         //Defaults
456         rightAreaMargin = 1;
457         leftAreaMargin = 1;
458         topAreaMargin = 1;
459         bottomAreaMargin = 1;
460             
461         //Shrink width to maintain aspect ratio
462         if (relevantHeight * aspectRatio <= relevantWidth){
463             double preferredWidth = relevantHeight * aspectRatio;
464             int marginSize = (int) (relevantWidth - preferredWidth) / 2;
465             
466             rightAreaMargin = 1 + marginSize;
467             leftAreaMargin = 1 + marginSize;
468         }
469         //Shrink height to maintain aspect ratio
470         else {
471             double preferredHeight = relevantWidth / aspectRatio;
472             int marginSize = (int) (relevantHeight - preferredHeight) / 2;
473             
474             topAreaMargin = 1 + marginSize;
475             bottomAreaMargin = 1 + marginSize;
476         }
477     }
