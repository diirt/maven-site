60      @Override
61      public void update(IntensityGraph2DRendererUpdate update) {
62          super.update(update);
63          if(update.getDrawLegend() != null){
64              drawLegend = update.getDrawLegend();
65          }
66          if(update.getColorMap() != null){
67              colorMap = update.getColorMap();
68              colorMapInstance = null;
69          }
70          if(update.getZLabelMargin() != null){
71              zLabelMargin = update.getZLabelMargin();
72          }
73          if(update.getLegendWidth() != null){
74              legendWidth = update.getLegendWidth();
75          }
76          if(update.getGraphAreaToLegendMargin() != null){
77              graphAreaToLegendMargin = update.getGraphAreaToLegendMargin();
78          }
79          if(update.getRightMargin() != null){
80              originalRightMargin = update.getRightMargin();
81          }
82          if(update.getXPixelSelectionRange()!= null){
83              xPixelSelectionRange = update.getXPixelSelectionRange();
84          }
85          if(update.getYPixelSelectionRange()!= null){
86              yPixelSelectionRange = update.getYPixelSelectionRange();
87          }
88      }

121     /**
122      *Draws an intensity graph in the given graphics context, using the given data.
123      * All drawing is done within the bounds specified either at initialization or at update.
124      *  Different colorSchemes may be specified using the IntensityGraph2DRendererUpdate class, in combination with the update function. 
125      * @param graphBuffer Contains <code>imageBuffer</code> and <code>Graphics2D</code> objects used to perform drawing functions within draw.
126      * @param data can not be null
127      */
128     public void draw(GraphBuffer graphBuffer, Cell2DDataset data) {
129         //Use super class to draw basics of graph.
130         this.g = graphBuffer.getGraphicsContext();
131         GraphAreaData area = new GraphAreaData();
132         BufferedImage image = graphBuffer.getImage();
133         calculateRanges(data.getXRange(), data.getXRange(), data.getYRange(), data.getYRange());
134         area.setGraphBuffer(graphBuffer);
135         graphBuffer.drawBackground(backgroundColor);
136         calculateZRange(data.getStatistics().getRange(), data.getDisplayRange());
137         
138         // TODO: the calculation for leaving space for the legend is somewhat hacked
139         // Instead of actually having a nice calculation, we increase the margin
140         // to the right before using the standard calculateGraphArea
141         int areaRightPixel;
142         if(drawLegend){
143             calculateZLabels();
144             areaRightPixel = getImageWidth() - 1 - (graphAreaToLegendMargin + legendWidth + zLabelMargin + zLabelMaxWidth + rightMargin);
145         } else {
146             areaRightPixel = getImageWidth() - 1 - rightMargin;
147         }
148         area.setGraphArea(leftMargin, getImageHeight() - 1 - bottomMargin, areaRightPixel, topMargin);
149         area.setGraphPadding(leftAreaMargin, bottomAreaMargin, rightAreaMargin, topAreaMargin);
150         area.setLabelMargin(xLabelMargin, yLabelMargin);
151         area.setRanges(getXPlotRange(), xValueScale, getYPlotRange(), yValueScale);
152         area.prepareLabels(labelFont, labelColor);
153         area.prepareGraphArea(true, referenceLineColor);
154         area.drawGraphArea();
155         
156         /*Wait to calculate the coordinates of the legend labels till after yPlotCoordRange is calculated.
157         Allows for the use of yPlotCoordEnd/start in calculations.*/
158         if(drawLegend){
159             if (zReferenceValues != null) {
160                 int[] zRefCoords = new int[zReferenceValues.size()];
161                 if(zRefCoords.length == 1){
162                     zRefCoords[0] = area.areaTop;
163                 }
164                 else{
165                     for (int i = 0; i < zRefCoords.length; i++) {
166                         zRefCoords[i] = (int) scaledZ(zReferenceValues.getDouble(i), area.graphBottom, area.graphTop);
167                     }
168                 }
169                 zReferenceCoords = new ArrayInt(zRefCoords);
170             }
171         }
172         g.setRenderingHint(RenderingHints.KEY_STROKE_CONTROL, RenderingHints.VALUE_STROKE_PURE);
173         
174         
175         //Set color scheme
176         if (!optimizeColorScheme){
177             colorMapInstance = colorMap.createInstance(zPlotRange);
178         } else {
179             if (colorMapInstance == null || !Objects.equals(optimizedRange, zRange)) {
180                 colorMapInstance = colorMap.createInstance(zPlotRange);
181                 colorMapInstance = NumberColorMaps.optimize(colorMapInstance, zPlotRange);
182                 optimizedRange = zPlotRange;
183             }
184         }
185 
186 
187         double xStartGraph = super.xPlotCoordStart;
188         double yEndGraph = area.graphBottom;
189 
190         double yHeightTotal = area.graphBottom - area.graphTop + 1;
191         
192         int startX = area.graphLeft;//(int) Math.floor(xPlotCoordStart);
193         int startY = area.graphTop;//(int) Math.floor(yPlotCoordStart);
194         int endX = area.graphRight;//(int) Math.ceil(xPlotCoordEnd);
195         int endY = area.graphBottom;//(int) Math.ceil(yPlotCoordEnd);
196         PointToDataMap xPointToDataMap = createXPointToDataMap(startX, endX, graphBuffer, data.getXBoundaries()); //createPointToDataMap(startX, endX+1, getXPlotRange(), data.getXBoundaries(), false);
197         PointToDataMap yPointToDataMap = createYPointToDataMap(startY, endY, graphBuffer, data.getYBoundaries());//getYPlotRange(), data.getYBoundaries(), true);
198         graphBuffer.drawDataImage(xPointToDataMap.startPoint, yPointToDataMap.startPoint, xPointToDataMap.pointToDataMap, yPointToDataMap.pointToDataMap, data, colorMapInstance);
199         
200         if(drawLegend && legendWidth>0){
201             /*dataList is made by splitting the aggregated range of the z(color) data into a list of the
202             same length as the the height of the graph in pixels.*/
203             ListNumber dataList = ListNumbers.linearListFromRange(zPlotRange.getMinimum(),zPlotRange.getMaximum(),(int)yHeightTotal);
204             //legendData is a Cell2DDataset representation of dataList.
205             Cell2DDataset legendData = Cell2DDatasets.linearRange(dataList, Ranges.range(0, 1), 1, Ranges.range(0, (int)yHeightTotal), (int)yHeightTotal);
206             int xLegendStart = getImageWidth() - originalRightMargin - zLabelMaxWidth - zLabelMargin - legendWidth;
207             drawRectanglesArray(g, legendData, xLegendStart, yEndGraph, legendWidth, yHeightTotal, 1, legendWidth, image);
208             graphBuffer.drawLeftLabels(zReferenceLabels, zReferenceCoords, labelColor, labelFont, area.areaBottom, area.areaTop, getImageWidth() - originalRightMargin - 1);
209         }
210         
211         // Calculate selection and draw rectangle
212         if (xPixelSelectionRange != null && yPixelSelectionRange != null) {
213             // Calculate the selection pixel box
214             int selectionLeftPixel = (int) xPixelSelectionRange.getMinimum();
215             int selectionRightPixel = (int) xPixelSelectionRange.getMaximum();
216             int selectionTopPixel = (int) yPixelSelectionRange.getMinimum();
217             int selectionBottomPixel = (int) yPixelSelectionRange.getMaximum();
218             
219             // Calculate the selection value range
220             double selectionLeftValue = graphBuffer.xPixelLeftToValue(selectionLeftPixel);
221             double selectionRightValue = graphBuffer.xPixelRightToValue(selectionRightPixel);
222             double selectionTopValue = graphBuffer.yPixelTopToValue(selectionTopPixel);
223             double selectionBottomValue = graphBuffer.yPixelBottomToValue(selectionBottomPixel);
224             xValueSelectionRange = Ranges.range(selectionLeftValue, selectionRightValue);
225             yValueSelectionRange = Ranges.range(selectionBottomValue, selectionTopValue);
226             
227             // Calcualte the selection data index boundaries
228             int xLeftOffset = selectionLeftPixel - xPointToDataMap.startPoint;
229             int xRightOffset = selectionRightPixel - xPointToDataMap.startPoint;
230             if ((xLeftOffset < 0 && xRightOffset <0) 
231                     || (xLeftOffset >= xPointToDataMap.pointToDataMap.length && xRightOffset >= xPointToDataMap.pointToDataMap.length)) {
232                 // Selection range is outside range
233                 xIndexSelectionRange = null;
234             } else {
235                 xLeftOffset = Math.max(0, xLeftOffset);
236                 xRightOffset = Math.min(xPointToDataMap.pointToDataMap.length, xRightOffset);
237                 xIndexSelectionRange = Ranges.range(xPointToDataMap.pointToDataMap[xLeftOffset], xPointToDataMap.pointToDataMap[xRightOffset]);
238             }
239             
240             int yBottomOffset = selectionBottomPixel - yPointToDataMap.startPoint;
241             int yTopOffset = selectionTopPixel - yPointToDataMap.startPoint;
242             if ((yBottomOffset < 0 && yTopOffset <0) 
243                     || (yBottomOffset >= yPointToDataMap.pointToDataMap.length && yTopOffset >= yPointToDataMap.pointToDataMap.length)) {
244                 // Selection range is outside range
245                 yIndexSelectionRange = null;
246             } else {
247                 yTopOffset = Math.max(0, yTopOffset);
248                 yBottomOffset = Math.min(xPointToDataMap.pointToDataMap.length, yBottomOffset);
249                 yIndexSelectionRange = Ranges.range(yPointToDataMap.pointToDataMap[yBottomOffset], yPointToDataMap.pointToDataMap[yTopOffset]);
250             }
251             
252             g.setRenderingHint(RenderingHints.KEY_STROKE_CONTROL, RenderingHints.VALUE_STROKE_NORMALIZE);
253             g.setColor(Color.BLACK);
254             graphBuffer.getGraphicsContext().drawRect(selectionLeftPixel, selectionTopPixel,
255                     selectionRightPixel - selectionLeftPixel, selectionBottomPixel - selectionTopPixel);
256         }
257     }

269     PointToDataMap createXPointToDataMap(int leftPixel, int rightPixel, GraphBuffer buffer, ListNumber xBoundaries) {
270         // Check any data in range
271         int minValuePixel = buffer.xValueToPixel(xBoundaries.getDouble(0));
272         int maxValuePixel = buffer.xValueToPixel(xBoundaries.getDouble(xBoundaries.size() - 1));
273         if ( minValuePixel > rightPixel ||
274                 maxValuePixel < leftPixel) {  
275             PointToDataMap result = new PointToDataMap();
276             result.pointToDataMap = new int[0];
277             result.startPoint = leftPixel;
278             return result;
279         }
280 
281         // Find the subset that fits
282         int startPixel = Math.max(minValuePixel, leftPixel);
283         int endPixel = Math.min(maxValuePixel, rightPixel);
284         int nPoints = endPixel - startPixel + 1;
285         int[] pointToDataMap = new int[nPoints];
286         
287         int currentValueIndex = 0;
288         int currentLeftBoundaryPixel = minValuePixel;
289         int currentRightBoundaryPixel = buffer.xValueToPixel(xBoundaries.getDouble(1));
290         
291         for (int currentOffset = 0; currentOffset < pointToDataMap.length; currentOffset++) {
292             // Advance if the pixel is past the right boundary, or if it's the same
293             // but the left boundary is passed (to "encourage" the change of value
294             // from large cells)
295             int currentPixel = startPixel + currentOffset;
296             while ((currentRightBoundaryPixel < currentPixel || (currentRightBoundaryPixel == currentPixel && currentLeftBoundaryPixel != currentPixel))
297                     && currentValueIndex < xBoundaries.size() - 2) {  
298                 currentValueIndex++;
299                 currentLeftBoundaryPixel = currentRightBoundaryPixel;
300                 currentRightBoundaryPixel = buffer.xValueToPixel(xBoundaries.getDouble(currentValueIndex + 1));
301             }
302             
303             pointToDataMap[currentOffset] = currentValueIndex;
304         }
305         
306         PointToDataMap result = new PointToDataMap();
307         result.pointToDataMap = pointToDataMap;
308         result.startPoint = startPixel;
309         return result;
310     }

312     PointToDataMap createYPointToDataMap(int topPixel, int bottomPixel, GraphBuffer buffer, ListNumber yBoundaries) {
313         // Check any data in range
314         int minValuePixel = buffer.yValueToPixel(yBoundaries.getDouble(0));
315         int maxValuePixel = buffer.yValueToPixel(yBoundaries.getDouble(yBoundaries.size() - 1));
316         if (minValuePixel < topPixel ||
317                 maxValuePixel > bottomPixel) {
318             PointToDataMap result = new PointToDataMap();
319             result.pointToDataMap = new int[0];
320             result.startPoint = topPixel;
321             return result;
322         }
323 
324         // Find the subset that fits
325         int startPixel = Math.max(maxValuePixel, topPixel);
326         int endPixel = Math.min(minValuePixel, bottomPixel);
327         int nPoints = endPixel - startPixel + 1;
328         int[] pointToDataMap = new int[nPoints];
329         
330         int currentValueIndex = 0;
331         int currentBottomBoundaryPixel = minValuePixel;
332         int currentTopBoundaryPixel = buffer.yValueToPixel(yBoundaries.getDouble(1));
333         
334         for (int currentOffset = 0; currentOffset < pointToDataMap.length; currentOffset++) {
335             // Advance if the pixel is past the top boundary, or if it's the same
336             // but the bottom boundary is passed (to "encourage" the change of value
337             // from large cells)
338             int currentPixel = endPixel - currentOffset;
339             while ((currentTopBoundaryPixel > currentPixel || (currentTopBoundaryPixel == currentPixel && currentBottomBoundaryPixel != currentPixel))
340                 && currentValueIndex < yBoundaries.size() - 2) {
341                 currentValueIndex++;
342                 currentBottomBoundaryPixel = currentTopBoundaryPixel;
343                 currentTopBoundaryPixel = buffer.yValueToPixel(yBoundaries.getDouble(currentValueIndex + 1));
344             }
345             
346             pointToDataMap[nPoints - currentOffset - 1] = currentValueIndex;
347         }
348         
349         PointToDataMap result = new PointToDataMap();
350         result.pointToDataMap = pointToDataMap;
351         result.startPoint = startPixel;
352         return result;
353     }

355     private void drawRectanglesArray(Graphics2D g, Cell2DDataset data, double xStartGraph, double yEndGraph,
356             double xWidthTotal, double yHeightTotal, double cellHeight, double cellWidth, BufferedImage image){
357         
358         byte pixels[] = ((DataBufferByte)image.getRaster().getDataBuffer()).getData();
359         boolean hasAlphaChannel = image.getAlphaRaster() != null;
360         int countY = 0;
361         int countX;
362         double yPosition = yEndGraph-yHeightTotal;
363         int yPositionInt = (int)(yEndGraph-yHeightTotal);
364         while (countY < data.getYCount()){
365                 countX = 0;
366                 double xPosition = xStartGraph;
367                 int xPositionInt = (int)xStartGraph;
368                 while (countX < data.getXCount()){
369                     int rgb = colorMapInstance.colorFor(data.getValue((int)countX, data.getYCount()-1-(int)countY));
370                     for(int w = 0; w < (int)cellWidth + 1; w++){
371                         for(int h = 0; h < (int)cellHeight + 1; h++){
372                             if(hasAlphaChannel){
373                             pixels[(yPositionInt+h)*getImageWidth()*4 + 4*xPositionInt + 0] = (byte)(rgb >> 24 & 0xFF);
374                             pixels[(yPositionInt+h)*getImageWidth()*4 + 4*xPositionInt + 1] = (byte)(rgb & 0xFF);
375                             pixels[(yPositionInt+h)*getImageWidth()*4 + 4*xPositionInt + 2] = (byte)(rgb >> 8 & 0xFF);
376                             pixels[(yPositionInt+h)*getImageWidth()*4 + 4*xPositionInt + 3] = (byte)(rgb >> 16 & 0xFF);
377                             }
378                             else{
379                             pixels[(yPositionInt+h)*getImageWidth()*3 + 3*(xPositionInt+w) + 0] = (byte)(rgb & 0xFF);
380                             pixels[(yPositionInt+h)*getImageWidth()*3 + 3*(xPositionInt+w) + 1] = (byte)((rgb >> 8 & 0xFF) );
381                             pixels[(yPositionInt+h)*getImageWidth()*3 + 3*(xPositionInt+w) + 2] = (byte)((rgb >> 16 & 0xFF));
382                             }
383                         }
384                     }
385                     xPosition = xPosition + cellWidth;
386                     xPositionInt = (int)xPosition;
387                     countX++;
388                 }
389                 yPosition = yPosition + cellHeight;
390                 yPositionInt = (int)yPosition;
391                 countY++;
392             }
393     }

395     /*Calculates the range of the z values to be graphed, based on the previous z range (if there is one)
396      If there is a previous range, the minimum value can only be lowered and the maximum value can only
397      be raised to match the current range.*/
398     /**
399      *Calculates the range of the z values to be graphed based on the previous z range (if there is one).
400      *  If there is a previous range, the minimum value can only be lowered and the maximum value can only
401      *be raised to match the current range.
402      * @param zDataRange current data range
403      * @param displayRange current display range
404      */
405     protected void calculateZRange(Range zDataRange, Range displayRange) {
406         zPlotRange = zAxisRange.axisRange(zDataRange, displayRange);
407     }
408     /**
409      *Sets private variables to account for the space required to draw in labels for the legend.
410      * Only called if drawLegend is true.
411      */
412     protected void calculateZLabels() {
413         labelFontMetrics = g.getFontMetrics(labelFont);
414         // Calculate z axis references. If range is zero, use special logic
415         if (!(zPlotRange.getMinimum() == zPlotRange.getMaximum())) {
416             ValueAxis zAxis = zValueScale.references(zPlotRange, 2, Math.max(2, getImageHeight() / 60));
417             zReferenceLabels = Arrays.asList(zAxis.getTickLabels());
418             zReferenceValues = new ArrayDouble(zAxis.getTickValues());            
419         } else {
420             // TODO: use something better to format the number
421             ValueAxis zAxis = zValueScale.references(zPlotRange, 1, 1);
422             zReferenceLabels = Arrays.asList(zAxis.getTickLabels());
423             zReferenceValues = new ArrayDouble(zPlotRange.getMinimum());            
424         }
425         
426         // Compute z axis spacing
427         int[] zLabelWidths = new int[zReferenceLabels.size()];
428         zLabelMaxWidth = 0;
429         for (int i = 0; i < zLabelWidths.length; i++) {
430             zLabelWidths[i] = labelFontMetrics.stringWidth(zReferenceLabels.get(i));
431             zLabelMaxWidth = Math.max(zLabelMaxWidth, zLabelWidths[i]);
432         }
433     }

435     /**
436      *Translates a value's position in the aggregated range to a position on the legend.
437      * @param value raw value
438      * @param bottomPixel the first pixel from the bottom
439      * @param topPixel the first pixel from the top
440      * @return double(scaled value)
441      */
442     protected final double scaledZ(double value, int bottomPixel, int topPixel) {
443         return Math.ceil(zValueScale.scaleValue(value, zPlotRange.getMinimum(), zPlotRange.getMaximum(), bottomPixel, topPixel));
444     }
