65      /**
66       * Changes the buffer where the graph area is going to be rendered.
67       * 
68       * @param graphBuffer a graph buffer
69       */
70      public void setGraphBuffer(GraphBuffer graphBuffer) {
71          this.graphBuffer = graphBuffer;

74      /**
75       * Change the portion of the buffer allocated to displaying the graph.
76       * It gives the range of pixels (inclusive of both sides) where the
77       * graph will be displayed.
78       * <p>
79       * The coordinate system is that of a standard image, where (0,0) is the
80       * top left corner.
81       * 
82       * @param areaLeft the first pixel on the left (inclusive)
83       * @param areaBottom the first pixel on the bottom (inclusive)
84       * @param areaRight the last pixel on the right (inclusive)
85       * @param areaTop the last pixel on the top (inclusive)
86       */
87      public void setGraphArea(int areaLeft, int areaBottom, int areaRight, int areaTop) {
88          this.areaLeft = areaLeft;
89          this.areaBottom = areaBottom;
90          this.areaRight = areaRight;
91          this.areaTop = areaTop;
92      }

94      /**
95       * Changes the padding of the graph area, effectively moving the displayed
96       * value range further into the graph. This is useful if value points
97       * are represented by a glyph, and one wants to leave space so that the
98       * glyph is not cropped. The space is still on the graph area, so reference
99       * lines and other points right outside the value range will be displayed.
100      * 
101      * @param graphPaddingLeft the number of pixels to be left to the left of the values
102      * @param graphPaddingBottom the number of pixels to be left to the bottom of the values
103      * @param graphPaddingRight the number of pixels to be left to the right of the values
104      * @param graphPaddingTop the number of pixels to be left to the top of the values
105      */
106     public void setGraphPadding(int graphPaddingLeft, int graphPaddingBottom, int graphPaddingRight, int graphPaddingTop) {
107         this.graphPaddingLeft = graphPaddingLeft;
108         this.graphPaddingBottom = graphPaddingBottom;
109         this.graphPaddingRight = graphPaddingRight;
110         this.graphPaddingTop = graphPaddingTop;
111     }
112     
113     /**
114      * Changes the margin between the labels and the graph area. This are
115      * is left blank, and is not part of the graph area.
116      * 
117      * @param labelMarginBottom margin in pixel between the bottom labels and the bottom part of the graph
118      * @param labelMarginLeft margin in pixel between the left labels and the left part of the graph
119      */
120     public void setLabelMargin(int labelMarginBottom, int labelMarginLeft) {
121         this.labelMarginBottom = labelMarginBottom;
122         this.labelMarginLeft = labelMarginLeft;
123     }

125     /**
126      * Sets the ranges of the values that are going to be displayed.
127      * <p>
128      * The value ranges need for a series of calculation before it is actually
129      * determined to what pixels they correspond in the graph area.
130      * 
131      * @param xValueRange the ranges of values on the horizontal axis
132      * @param xValueScale the scale for the horizontal axis
133      * @param yValueRange the ranges of values on the vertical axis
134      * @param yValueScale the scale for the vertical axis
135      */
136     public void setRanges(Range xValueRange, ValueScale xValueScale, Range yValueRange, ValueScale yValueScale) {
137         this.xValueRange = xValueRange;
138         this.yValueRange = yValueRange;
139         this.xValueScale = xValueScale;
140         this.yValueScale = yValueScale;
141     }
142     
143     /**
144      * Prepares the label text and values for both the bottom and left axis.
145      * 
146      * @param labelFont the font for the label
147      * @param labelColor the color for the label
148      */
149     public void prepareLabels(Font labelFont, Color labelColor) {
150         this.labelFont = labelFont;
151         this.labelColor = labelColor;
152         
153         // Calculate horizontal axis references. If range is zero, use special logic
154         if (!(xValueRange.getMinimum() == xValueRange.getMaximum())) {
155             ValueAxis xAxis = xValueScale.references(xValueRange, 2, Math.max(2, (areaRight - areaLeft + 1) / 55));
156             xReferenceLabels = Arrays.asList(xAxis.getTickLabels());
157             xReferenceValues = new ArrayDouble(xAxis.getTickValues());            
158         } else {
159             // TODO: use something better to format the number
160             xReferenceLabels = Collections.singletonList(Double.toString(xValueRange.getMinimum()));
161             xReferenceValues = new ArrayDouble(xValueRange.getMinimum());  
162         }      
163         
164         // Calculate vertical axis references. If range is zero, use special logic
165         if (!(yValueRange.getMinimum() == yValueRange.getMaximum())) {
166             ValueAxis yAxis = yValueScale.references(yValueRange, 2, Math.max(2, (areaBottom - areaTop + 1) / 55));
167             yReferenceLabels = Arrays.asList(yAxis.getTickLabels());
168             yReferenceValues = new ArrayDouble(yAxis.getTickValues());
169         } else {
170             // TODO: use something better to format the number
171             yReferenceLabels = Collections.singletonList(Double.toString(yValueRange.getMinimum()));
172             yReferenceValues = new ArrayDouble(yValueRange.getMinimum());  
173         }
174         
175         FontMetrics labelFontMetrics = graphBuffer.getGraphicsContext().getFontMetrics(labelFont);
176         
177         // Compute x axis spacing
178         xLabelMaxHeight = labelFontMetrics.getHeight() - labelFontMetrics.getLeading();
179         
180         // Compute y axis spacing
181         int[] yLabelWidths = new int[yReferenceLabels.size()];
182         yLabelMaxWidth = 0;
183         for (int i = 0; i < yLabelWidths.length; i++) {
184             yLabelWidths[i] = labelFontMetrics.stringWidth(yReferenceLabels.get(i));
185             yLabelMaxWidth = Math.max(yLabelMaxWidth, yLabelWidths[i]);
186         }
187     }
188     
189     /**
190      * If the range is zero, fake a range.
191      * 
192      * @param range a range
193      * @return the same range, or one that is safe to draw
194      */
195     private Range safeRange(Range range) {
196         if (range.getMinimum() == range.getMaximum()) {
197             return Ranges.range(range.getMinimum() - 1.0, range.getMaximum() + 1.0);
198         } else {
199             return range;
200         }
201     }
202     
203     /**
204      * Final computation to prepare pixel position of graph area and references.
205      * 
206      * @param asCell whether the graph area should represent cells or points
207      */
208     public void prepareGraphArea(boolean asCell, Color referenceLineColor) {
209         this.referenceLineColor = referenceLineColor;
210         
211         // Prepare x positions
212         graphLeft = areaLeft + yLabelMaxWidth + labelMarginLeft;
213         graphRight = areaRight;
214         if (asCell) {
215             graphBuffer.setXScaleAsCell(safeRange(xValueRange), graphLeft + graphPaddingLeft, graphRight - graphPaddingRight, xValueScale);
216         } else {
217             graphBuffer.setXScaleAsPoint(safeRange(xValueRange), graphLeft + graphPaddingLeft, graphRight - graphPaddingRight, xValueScale);
218         }
219         
220         // Prepare y positions
221         graphTop = areaTop;
222         graphBottom = areaBottom - xLabelMaxHeight - labelMarginBottom;
223         if (asCell) {
224             graphBuffer.setYScaleAsCell(safeRange(yValueRange), graphBottom - graphPaddingBottom, graphTop + graphPaddingTop, yValueScale);
225         } else {
226             graphBuffer.setYScaleAsPoint(safeRange(yValueRange), graphBottom - graphPaddingBottom, graphTop + graphPaddingTop, yValueScale);
227         }
228         
229         //Only calculates reference coordinates if calculateLabels() was called
230         if (xReferenceValues != null) {
231             int[] xRefCoords = new int[xReferenceValues.size()];
232             for (int i = 0; i < xRefCoords.length; i++) {
233                 xRefCoords[i] = graphBuffer.xValueToPixel(xReferenceValues.getDouble(i));
234             }
235             if (asCell && xRefCoords[xReferenceValues.size() - 1] == graphRight + 1) {
236                 xRefCoords[xReferenceValues.size() - 1]--;
237             }
238             xReferencePixels = new ArrayInt(xRefCoords);
239         }
240         
241         if (yReferenceValues != null) {
242             int[] yRefCoords = new int[yReferenceValues.size()];
243             for (int i = 0; i < yRefCoords.length; i++) {
244                 yRefCoords[i] = graphBuffer.yValueToPixel(yReferenceValues.getDouble(i));
245             }
246             if (asCell && yRefCoords[yReferenceValues.size() - 1] == graphTop - 1) {
247                 yRefCoords[yReferenceValues.size() - 1]++;
248             }
249             yReferencePixels = new ArrayInt(yRefCoords);
250         }
251     }
252     
253     protected void drawGraphArea() {
254         graphBuffer.getGraphicsContext().setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
255         
256         graphBuffer.drawVerticalReferenceLines(xReferencePixels, referenceLineColor, graphBottom, graphTop);
257         graphBuffer.drawHorizontalReferenceLines(yReferencePixels, referenceLineColor, graphLeft, graphRight);
258         
259         graphBuffer.drawBottomLabels(xReferenceLabels, xReferencePixels, labelColor, labelFont, graphLeft, graphRight, graphBottom + labelMarginBottom + 1);
260         graphBuffer.drawLeftLabels(yReferenceLabels, yReferencePixels, labelColor, labelFont, graphBottom, graphTop, graphLeft - labelMarginLeft - 1);
261     }
