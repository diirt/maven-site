13  public class MathUtil {
14      
15      /**
16       *Returns the base 10 logarithm of the given value, to the nearest integer value.
17       * <p>Uses Math.log10(double a)</p>
18       * <p>Special cases: (from Math.log10(double a))</p>
19       * <ul>
20       *  <li>If the argument is NaN or less than zero, then the result is NaN.</li>
21       *  <li>If the argument is positive infinity, then the result is positive infinity.</li>
22       *  <li>If the argument is positive zero or negative zero, then the result is negative infinity.</li>
23       *  <li>If the argument is equal to 10n for integer n, then the result is n</li>
24       * </ul>
25       * @param value double
26       * @return truncated base 10 logarithm of the given value.
27       */
28      public static int orderOf(double value) {
29          return (int) Math.floor(Math.log10(value));
30      }
31  
32      /**
33       * Returns the ratio of the given value to the total range as a <code>double</code> between 0 and 1.
34       *
35       * @param value - number between min and max, inclusive.
36       * @param min - lowest number in range.
37       * @param max - highest number in range.
38       * @return <code>double</code> from 0 to 1, inclusive.
39       */
40      public static double normalize(double value, double min, double max) {
41          if (value < min || value > max) {
42              throw new IllegalArgumentException("value not in range");
43          }
44          return (value - min) / (max - min);
45      }
46  
47      /**
48       * Returns the ratio of the given value to the total range as a <code>double</code> between 0 and 1, multiplied by newWidth.
49       *
50       * @param value - number between min and max, inclusive.
51       * @param min - lowest number in range.
52       * @param max - highest number in range.
53       * @param newWidth - scalar to multiply the ratio by.
54       * @return <code>double</code> from 0 to newWidth, inclusive.
55       */
56      public static double scale(double value, double min, double max, double newWidth) {
57          return (value - min) * newWidth / (max - min);
58      }
59  }
