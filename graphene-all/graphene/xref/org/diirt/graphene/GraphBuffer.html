22  /**
23   * Provides low level drawing operations, at the right granularity to perform
24   * them efficiently, but without performing any high level decisions (e.g. formatting,
25   * layout)
26   * <p>
27   * This class takes care of putting the pixels where it is told. It specifically
28   * does not concerns itself with the calculation of what/where to draw. It provides
29   * the drawing of aggregated data structures, so that the plotting can be 
30   * efficient and clean.
31   * <p>
32   * It also serves as a wrapper around Java2D (<code>Graphics2D</code>) so that
33   * the drawing can be re-implemented efficiently on other engines in the future,
34   * such as JavaFX.
35   * 
36   * @author carcassi, sjdallst, asbarber, jkfeng
37   */
38  public class GraphBuffer {

51      /**
52       * Creates a GraphBuffer with the given image on which to draw a graph.
53       * 
54       * @param image an image on which we can draw a graph
55       */
56      private GraphBuffer(BufferedImage image){
57          this.image = image;
58          width = image.getWidth();
59          height = image.getHeight();
60          pixels = ((DataBufferByte)this.image.getRaster().getDataBuffer()).getData();
61          hasAlphaChannel = image.getAlphaRaster() != null;
62          g = image.createGraphics();
63      }

85      /**
86       * Changes the pixel at the given coordinates to the given color.
87       * TODO: make sure all other plotting functions use this.
88       * 
89       * @param x	x-coordinate of a pixel
90       * @param y	y-coordinate of a pixel
91       * @param color color-value of the pixel
92       */
93      public void setPixel(int x, int y, int color){
94          if(hasAlphaChannel){
95              pixels[y*image.getWidth()*4 + x*4 + 3] = (byte)(color >> 24 & 0xFF);
96              pixels[y*image.getWidth()*4 + x*4 + 0] = (byte)(color >> 0 & 0xFF);
97              pixels[y*image.getWidth()*4 + x*4 + 1] = (byte)(color >> 8 & 0xFF);
98              pixels[y*image.getWidth()*4 + x*4 + 2] = (byte)(color >> 16 & 0xFF);
99          }
100         else{
101             pixels[y*image.getWidth()*3 + x*3 + 0] = (byte)(color >> 0 & 0xFF);
102             pixels[y*image.getWidth()*3 + x*3 + 1] = (byte)(color >> 8 & 0xFF);
103             pixels[y*image.getWidth()*3 + x*3 + 2] = (byte)(color >> 16 & 0xFF);
104         }
105     }

127     /**
128      * Plots a two dimensional array of values encoded by color. 
129      * 
130      * @param xStartPoint the horizontal coordinate for the first pixel of the image
131      * @param yStartPoint the vertical coordinate for the first pixel of the image
132      * @param xPointToDataMap a map from pixel horizontal offset to data index
133      * @param yPointToDataMap a map from pixel vertical offset to data index
134      * @param data the dataset to be plotted
135      * @param colorMap the color map
136      */
137     public void drawDataImage(int xStartPoint, int yStartPoint,
138 			int[] xPointToDataMap, int[] yPointToDataMap,
139 			Cell2DDataset data, NumberColorMapInstance colorMap) {
140         int previousYData = -1;
141 	
142 	// Loop through the points to be plotted. The length of the image is
143         // given by the point to data map, since it tells how many points are mapped.
144         for (int yOffset = 0; yOffset < yPointToDataMap.length; yOffset++) {
145             int yData = yPointToDataMap[yOffset];
146             if (yData != previousYData) {
147                 for (int xOffset = 0; xOffset < xPointToDataMap.length; xOffset++) {  
148                     int xData = xPointToDataMap[xOffset];
149 		    
150                     // Get the value, convert to color and plot
151                     int rgb = colorMap.colorFor(data.getValue(xData, yData));
152 		    setPixel( xStartPoint + xOffset  , yStartPoint + yOffset , rgb );
153                 }
154 		
155 	    // If the current line is the same as the previous, it's
156             // faster to make a copy
157             } else {
158                 if (hasAlphaChannel) {
159                     System.arraycopy(pixels, (yStartPoint + yOffset - 1)*width*4 + 4*xStartPoint,
160                             pixels, (yStartPoint + yOffset)*width*4 + 4*xStartPoint, xPointToDataMap.length*4);
161                 } else {
162                     System.arraycopy(pixels, (yStartPoint + yOffset - 1)*width*3 + 3*xStartPoint,
163                             pixels, (yStartPoint + yOffset)*width*3 + 3*xStartPoint, xPointToDataMap.length*3);
164                 }
165             }
166             previousYData = yData;
167         }
168     }

349     /**
350      * Draws the given labels a the bottom of the graph area.
351      * <p>
352      * This method may not display some labels in case they would overlap with
353      * each other. It does try, though, to make sure that the first and the
354      * last label are always displayed.
355      * 
356      * @param labels a list of x-axis labels to be drawn
357      * @param valuePixelPositions the central x-coordinate of each label
358      * @param labelColor color of the label text
359      * @param labelFont font style of the label text
360      * @param leftPixel the leftmost x-coordinate at which the label may be drawn
361      * @param rightPixel the rightmost x-coordinate at which the label may be drawn
362      * @param topPixel the y-coordinate of the top of the label
363      */
364     void drawBottomLabels(List<String> labels, ListInt valuePixelPositions, Color labelColor, Font labelFont, int leftPixel, int rightPixel, int topPixel) {
365         // Draw X labels
366         if (labels != null && !labels.isEmpty()) {
367             //g.setRenderingHint(RenderingHints.KEY_STROKE_CONTROL, RenderingHints.VALUE_STROKE_PURE);
368             g.setColor(labelColor);
369             g.setFont(labelFont);
370             FontMetrics metrics = g.getFontMetrics();
371 
372             // Draw first and last label to make sure they fit
373             int[] drawRange = new int[] {leftPixel, rightPixel};
374             drawLineLabel(g, metrics, labels.get(0), valuePixelPositions.getInt(0),
375                 drawRange, topPixel, true, false);
376             drawLineLabel(g, metrics, labels.get(labels.size() - 1),
377                     valuePixelPositions.getInt(labels.size() - 1),
378                 drawRange, topPixel, false, false);
379             
380             for (int i = 1; i < labels.size() - 1; i++) {
381                 drawLineLabel(g, metrics, labels.get(i), valuePixelPositions.getInt(i),
382                     drawRange, topPixel, true, false);
383             }
384         }

387     /**
388      * Draws some labels, which are plaintext, on the graph.
389      * 
390      * @param graphics the graphics object containing the graph
391      * @param metrics the font style
392      * @param text the text of the label
393      * @param xCenter x coordinate of where the label should be centered
394      * @param drawRange defines the range of coordinates where this label is allowed to be centered
395      * @param yTop define the y-coordinate of the top of the label
396      * @param updateMin whether we should be updating the minimum of the draw range
397      * or the maximum
398      * @param centeredOnly true if the label should be only displayed if it can be
399      * properly centered, and skipped if it cannot
400      */
401     private static void drawLineLabel(Graphics2D graphics, FontMetrics metrics, String text, int xCenter, int[] drawRange, int yTop, boolean updateMin, boolean centeredOnly) {
402         // If the center is not in the range, don't draw anything
403         if (drawRange[MAX] < xCenter || drawRange[MIN] > xCenter)
404             return;
405         
406         // If there is no space, don't draw anything
407         if (drawRange[MAX] - drawRange[MIN] < metrics.getHeight())
408             return;
409         
410         Java2DStringUtilities.Alignment alignment = Java2DStringUtilities.Alignment.TOP;
411         int targetX = xCenter;
412         int halfWidth = metrics.stringWidth(text) / 2;
413         if (xCenter < drawRange[MIN] + halfWidth) {
414             // Can't be drawn in the center
415             if (centeredOnly)
416                 return;
417             alignment = Java2DStringUtilities.Alignment.TOP_LEFT;
418             targetX = drawRange[MIN];
419         } else if (xCenter > drawRange[MAX] - halfWidth) {
420             // Can't be drawn in the center
421             if (centeredOnly)
422                 return;
423             alignment = Java2DStringUtilities.Alignment.TOP_RIGHT;
424             targetX = drawRange[MAX];
425         }
426 
427         Java2DStringUtilities.drawString(graphics, alignment, targetX, yTop, text);
428         
429         if (updateMin) {
430             drawRange[MIN] = targetX + metrics.getHeight();
431         } else {
432             drawRange[MAX] = targetX - metrics.getHeight();
433         }
434     }

458     private static void drawColumnLabel(Graphics2D graphics, FontMetrics metrics, String text, int yCenter, int[] drawRange, int xRight, boolean updateMin, boolean centeredOnly) {
459         // If the center is not in the range, don't draw anything
460         if (drawRange[MAX] < yCenter || drawRange[MIN] > yCenter)
461             return;
462         
463         // If there is no space, don't draw anything
464         if (drawRange[MAX] - drawRange[MIN] < metrics.getHeight())
465             return;
466         
467         Java2DStringUtilities.Alignment alignment = Java2DStringUtilities.Alignment.RIGHT;
468         int targetY = yCenter;
469         int halfHeight = metrics.getAscent() / 2;
470         if (yCenter < drawRange[MIN] + halfHeight) {
471             // Can't be drawn in the center
472             if (centeredOnly)
473                 return;
474             alignment = Java2DStringUtilities.Alignment.TOP_RIGHT;
475             targetY = drawRange[MIN];
476         } else if (yCenter > drawRange[MAX] - halfHeight) {
477             // Can't be drawn in the center
478             if (centeredOnly)
479                 return;
480             alignment = Java2DStringUtilities.Alignment.BOTTOM_RIGHT;
481             targetY = drawRange[MAX];
482         }
483 
484         Java2DStringUtilities.drawString(graphics, alignment, xRight, targetY, text);
485         
486         if (updateMin) {
487             drawRange[MAX] = targetY - metrics.getHeight();
488         } else {
489             drawRange[MIN] = targetY + metrics.getHeight();
490         }
491     }

518     public void drawValueLine(ListNumber xValues, ListNumber yValues, InterpolationScheme interpolation, ProcessValue pv) {
519         ReductionScheme reductionScheme = ReductionScheme.NONE;
520         
521         g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF);
522         g.setRenderingHint(RenderingHints.KEY_STROKE_CONTROL, RenderingHints.VALUE_STROKE_PURE);
523 
524         ScaledData scaledData;
525         
526         switch (reductionScheme) {
527             default:
528                 throw new IllegalArgumentException("Reduction scheme " + reductionScheme + " not supported");
529             case NONE:
530                 scaledData = scaleNoReduction(xValues, yValues,pv);
531                 break;
532         }
533         
534         // create path
535         Path2D path;
536         switch (interpolation) {
537             default:
538             case NEAREST_NEIGHBOR:
539                 path = nearestNeighbour(scaledData);
540                 break;
541             case LINEAR:
542                 path = linearInterpolation(scaledData);
543                 break;
544             case CUBIC:
545                 path = cubicInterpolation(scaledData);
546                 break;
547         }
548 
549         // Draw the line
550         g.draw(path);
551     }

560     private void drawValueExplicitLine(ListNumber xValues, ListNumber yValues, InterpolationScheme interpolation, ReductionScheme reduction,ProcessValue pv) {
561        
562         g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF);
563         g.setRenderingHint(RenderingHints.KEY_STROKE_CONTROL, RenderingHints.VALUE_STROKE_PURE);
564         
565         ScaledData scaledData;
566              
567         int start = ListNumbers.binarySearchValueOrLower(xValues, xLeftValue);
568         int end = ListNumbers.binarySearchValueOrHigher(xValues, xRightValue);
569         
570         xValues = ListMath.limit(xValues, start, end + 1);
571         yValues = ListMath.limit(yValues, start, end + 1);
572         
573         switch (reduction) {
574             default:
575                 throw new IllegalArgumentException("Reduction scheme " + reduction + " not supported");
576             case NONE:
577                 scaledData = scaleNoReduction(xValues, yValues, start,pv);
578                 break;
579             case FIRST_MAX_MIN_LAST:
580                 scaledData = scaleFirstMaxMinLastReduction(xValues, yValues, start,pv);
581                 break;
582         }
583         
584         // create path
585         Path2D path;
586         switch (interpolation) {
587             default:
588             case NEAREST_NEIGHBOR:
589                 path = nearestNeighbour(scaledData);
590                 break;
591             case LINEAR:
592                 path = linearInterpolation(scaledData);
593                 break;
594             case CUBIC:
595                 path = cubicInterpolation(scaledData);
596                 break;
597         }
598 
599         // Draw the line
600         g.draw(path);
601     }

603     private ScaledData scaleNoReduction(ListNumber xValues, ListNumber yValues,ProcessValue pv) {

607     private ScaledData scaleNoReduction(ListNumber xValues, ListNumber yValues, int dataStart,ProcessValue pv) {
608         ScaledData scaledData = new ScaledData();
609         int dataCount = xValues.size();
610         scaledData.scaledX = new double[dataCount];
611         scaledData.scaledY = new double[dataCount];
612         for (int i = 0; i < scaledData.scaledY.length; i++) {
613             scaledData.scaledX[i] = xValueToPixel(xValues.getDouble(i));
614             scaledData.scaledY[i] = yValueToPixel(yValues.getDouble(i));
615             pv.processScaledValue(dataStart + i, xValues.getDouble(i), yValues.getDouble(i), scaledData.scaledX[i], scaledData.scaledY[i]);
616         }
617         scaledData.end = dataCount;
618         return scaledData;
619     }

621     private ScaledData scaleFirstMaxMinLastReduction(ListNumber xValues, ListNumber yValues, int dataStart, ProcessValue pv) {
622         // The number of points generated by this is about 4 times the 
623         // number of points on the x axis. If the number of points is less
624         // than that, it's not worth it. Don't do the data reduction.
625         int xPlotCoordWidth=300; 
626         if (xValues.size() < xPlotCoordWidth * 4) {
627             return scaleNoReduction(xValues, yValues, dataStart,pv);
628         }
629 
630         ScaledData scaledData = new ScaledData();
631         scaledData.scaledX = new double[((int) xPlotCoordWidth + 1)*4 ];
632         scaledData.scaledY = new double[((int) xPlotCoordWidth + 1)*4];
633         int cursor = 0;
634         int previousPixel = xValueToPixel(xValues.getDouble(0));
635         double last = yValueToPixel(yValues.getDouble(0));
636         double min = last;
637         double max = last;
638         scaledData.scaledX[0] = previousPixel;
639         scaledData.scaledY[0] = min;
640         pv.processScaledValue(dataStart, xValues.getDouble(0), yValues.getDouble(0), xValueToPixel(xValues.getDouble(0)), last);
641         cursor++;
642         for (int i = 1; i < xValues.size(); i++) {
643             double currentScaledX = xValueToPixel(xValues.getDouble(i));
644             int currentPixel = (int) currentScaledX;
645             if (currentPixel == previousPixel) {
646                 last = yValueToPixel(yValues.getDouble(i));
647                 min = MathIgnoreNaN.min(min, last);
648                 max = MathIgnoreNaN.max(max, last);
649                 pv.processScaledValue(dataStart + i, xValues.getDouble(i), yValues.getDouble(i), currentScaledX, last);
650             } else {
651                 scaledData.scaledX[cursor] = previousPixel;
652                 scaledData.scaledY[cursor] = max;
653                 cursor++;
654                 scaledData.scaledX[cursor] = previousPixel;
655                 scaledData.scaledY[cursor] = min;
656                 cursor++;
657                 scaledData.scaledX[cursor] = previousPixel;
658                 scaledData.scaledY[cursor] = last;
659                 cursor++;
660                 previousPixel = currentPixel;
661                 last = yValueToPixel(yValues.getDouble(i));
662                 min = last;
663                 max = last;
664                 scaledData.scaledX[cursor] = currentPixel;
665                 scaledData.scaledY[cursor] = last;
666                 cursor++;
667             }
668         }
669         scaledData.scaledX[cursor] = previousPixel;
670         scaledData.scaledY[cursor] = max;
671         cursor++;
672         scaledData.scaledX[cursor] = previousPixel;
673         scaledData.scaledY[cursor] = min;
674         cursor++;
675         scaledData.end = cursor;
676         return scaledData;
677     }

679     private static Path2D.Double nearestNeighbour(ScaledData scaledData) {
680         double[] scaledX = scaledData.scaledX;
681         double[] scaledY = scaledData.scaledY;
682         int start = scaledData.start;
683         int end = scaledData.end;
684         Path2D.Double line = new Path2D.Double();
685         line.moveTo(scaledX[start], scaledY[start]);
686         for (int i = 1; i < end; i++) {
687             double halfX = scaledX[i - 1] + (scaledX[i] - scaledX[i - 1]) / 2;
688             if (!java.lang.Double.isNaN(scaledY[i-1])) {
689                 line.lineTo(halfX, scaledY[i - 1]);
690                 if (!java.lang.Double.isNaN(scaledY[i]))
691                     line.lineTo(halfX, scaledY[i]);
692             } else {
693                 line.moveTo(halfX, scaledY[i]);
694             }
695         }
696         line.lineTo(scaledX[end - 1], scaledY[end - 1]);
697         return line;
698     }

700     private static Path2D.Double linearInterpolation(ScaledData scaledData){
701         double[] scaledX = scaledData.scaledX;
702         double[] scaledY = scaledData.scaledY;
703         int start = scaledData.start;
704         int end = scaledData.end;
705         Path2D.Double line = new Path2D.Double();
706         
707         for (int i = start; i < end; i++) {
708             // Do I have a current value?
709             if (!java.lang.Double.isNaN(scaledY[i])) {
710                 // Do I have a previous value?
711                 if (i != start && !java.lang.Double.isNaN(scaledY[i - 1])) {
712                     // Here I have both the previous value and the current value
713                     line.lineTo(scaledX[i], scaledY[i]);
714                 } else {
715                     // Don't have a previous value
716                     // Do I have a next value?
717                     if (i != end - 1 && !java.lang.Double.isNaN(scaledY[i + 1])) {
718                         // There is no value before, but there is a value after
719                         line.moveTo(scaledX[i], scaledY[i]);
720                     } else {
721                         // There is no value either before or after
722                         line.moveTo(scaledX[i] - 1, scaledY[i]);
723                         line.lineTo(scaledX[i] + 1, scaledY[i]);
724                     }
725                 }
726             } 
727         }
728         return line;
729     }

731     private static Path2D.Double cubicInterpolation(ScaledData scaledData){
732         double[] scaledX = scaledData.scaledX;
733         double[] scaledY = scaledData.scaledY;
734         int start = scaledData.start;
735         int end = scaledData.end;
736         Path2D.Double path = new Path2D.Double();
737         for (int i = start; i < end; i++) {
738             
739             double y1;
740             double y2;
741             double x1;
742             double x2;
743             double y0;
744             double x0;
745             double y3;
746             double x3;
747             
748             double bx0;
749             double by0;
750             double bx3;
751             double by3;
752             double bdy0;
753             double bdy3;
754             double bx1;
755             double by1;
756             double bx2;
757             double by2;
758             
759             //Do I have current value?
760             if (!java.lang.Double.isNaN(scaledY[i])){
761                 //Do I have previous value?
762                 if (i > start && !java.lang.Double.isNaN(scaledY[i - 1])) {
763                     //Do I have value two before?
764                     if (i > start + 1 && !java.lang.Double.isNaN(scaledY[i - 2])) {
765                         //Do I have next value?
766                         if (i != end - 1 && !java.lang.Double.isNaN(scaledY[i + 1])) {
767                             y2 = scaledY[i];
768                             x2 = scaledX[i];
769                             y0 = scaledY[i - 2];
770                             x0 = scaledX[i - 2];
771                             y3 = scaledY[i + 1];
772                             x3 = scaledX[i + 1];
773                             y1 = scaledY[i - 1];
774                             x1 = scaledX[i - 1];
775                             bx0 = x1;
776                             by0 = y1;
777                             bx3 = x2;
778                             by3 = y2;
779                             bdy0 = (y2 - y0) / (x2 - x0);
780                             bdy3 = (y3 - y1) / (x3 - x1);
781                             bx1 = bx0 + (x2 - x0) / 6.0;
782                             by1 = (bx1 - bx0) * bdy0 + by0;
783                             bx2 = bx3 - (x3 - x1) / 6.0;
784                             by2 = (bx2 - bx3) * bdy3 + by3;
785                             path.curveTo(bx1, by1, bx2, by2, bx3, by3);
786                         } 
787                         else{//Have current, previous, two before, but not value after
788                             y2 = scaledY[i];
789                             x2 = scaledX[i];
790                             y1 = scaledY[i - 1];
791                             x1 = scaledX[i - 1];
792                             y0 = scaledY[i - 2];
793                             x0 = scaledX[i - 2];
794                             y3 = y2 + (y2 - y1) / 2;
795                             x3 = x2 + (x2 - x1) / 2;
796                             bx0 = x1;
797                             by0 = y1;
798                             bx3 = x2;
799                             by3 = y2;
800                             bdy0 = (y2 - y0) / (x2 - x0);
801                             bdy3 = (y3 - y1) / (x3 - x1);
802                             bx1 = bx0 + (x2 - x0) / 6.0;
803                             by1 = (bx1 - bx0) * bdy0 + by0;
804                             bx2 = bx3 - (x3 - x1) / 6.0;
805                             by2 = (bx2 - bx3) * bdy3 + by3;
806                             path.curveTo(bx1, by1, bx2, by2, bx3, by3);
807                         } 
808                     } else if (i != end - 1 && !java.lang.Double.isNaN(scaledY[i + 1])) {
809                         //Have current , previous, and next, but not two before
810                         path.moveTo(scaledX[i - 1], scaledY[i - 1]);
811                         y2 = scaledY[i];
812                         x2 = scaledX[i];
813                         y1 = scaledY[i - 1];
814                         x1 = scaledX[i - 1];
815                         y0 = y1 - (y2 - y1) / 2;
816                         x0 = x1 - (x2 - x1) / 2;
817                         y3 = scaledY[i + 1];
818                         x3 = scaledX[i + 1];
819                         bx0 = x1;
820                         by0 = y1;
821                         bx3 = x2;
822                         by3 = y2;
823                         bdy0 = (y2 - y0) / (x2 - x0);
824                         bdy3 = (y3 - y1) / (x3 - x1);
825                         bx1 = bx0 + (x2 - x0) / 6.0;
826                         by1 = (bx1 - bx0) * bdy0 + by0;
827                         bx2 = bx3 - (x3 - x1) / 6.0;
828                         by2 = (bx2 - bx3) * bdy3 + by3;
829                         path.curveTo(bx1, by1, bx2, by2, bx3, by3);
830                     }else{//have current, previous, but not two before or next
831                         path.lineTo(scaledX[i], scaledY[i]);
832                     }
833                 //have current, but not previous
834                 }else{
835                     // No previous value
836                     if (i != end - 1 && !java.lang.Double.isNaN(scaledY[i + 1])) {
837                         // If we have the next value, just move, we'll draw later
838                         path.moveTo(scaledX[i], scaledY[i]);
839                     } else {
840                         // If not, write a small horizontal line
841                         path.moveTo(scaledX[i] - 1, scaledY[i]);
842                         path.lineTo(scaledX[i] + 1, scaledY[i]);
843                     }
844                 }
845             }else{ //do not have current
846                // Do nothing
847              }
848         }
849         return path;
850     }
