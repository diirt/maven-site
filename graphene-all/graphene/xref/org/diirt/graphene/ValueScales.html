5   package org.diirt.graphene;
6   
7   /**
8    * The <code>ValueScales</code> class is useful to create
9    * different types of <code>ValueScale</code>s.
10   * It cannot be instantiated.
11   * 
12   * <p>Allows for a both a linear and logarithmic scale to be created.
13   * The <code>ValueScale</code> objects handle the scaling themselves.
14   * 
15   * @author carcassi
16   */
17  public class ValueScales {
18      
19      /**
20       * Cannot be instantiated.
21       */
22      private ValueScales(){
23      }
24      
25      /**
26       * Returns a linear scale to handle value scaling.
27       * @return linear scale
28       */
29      public static ValueScale linearScale() {
30          return new LinearValueScale();
31      }
32      
33      /**
34       * Returns a logarithmic scale to handle value scaling.
35       * @return logarithmic scale
36       */
37      public static ValueScale logScale() {
38          return new LogValueScale();
39      }
40  }
