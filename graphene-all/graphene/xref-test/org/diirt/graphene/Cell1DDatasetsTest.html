19  public class Cell1DDatasetsTest {
20  
21      @Test
22      public void linearRange1() {
23          ListNumber values = new ArrayDouble(5, 3, 7, -1, 2);
24          Cell1DDataset dataset = Cell1DDatasets.linearRange(values, 0, 10);
25          
26          assertThat(dataset.getXCount(), equalTo(5));
27          assertThat(dataset.getXRange().getMinimum(), equalTo(0.0));
28          assertThat(dataset.getXRange().getMaximum(), equalTo(10.0));
29          assertThat(dataset.getStatistics().getAverage(), equalTo(3.2));
30          assertThat(dataset.getStatistics().getStdDev(), closeTo(2.71293, 0.0001));
31          assertThat(dataset.getStatistics().getRange().getMinimum(), equalTo((Number) (-1.0)));
32          assertThat(dataset.getStatistics().getRange().getMaximum(), equalTo((Number) 7.0));
33          assertThat(dataset.getStatistics().getCount(), equalTo(5));
34          
35          // Check values
36          assertThat(dataset.getXBoundaries().getDouble(0), equalTo(0.0));
37          assertThat(dataset.getValue(0), equalTo(5.0));
38          assertThat(dataset.getXBoundaries().getDouble(1), equalTo(2.0));
39          assertThat(dataset.getValue(1), equalTo(3.0));
40          assertThat(dataset.getXBoundaries().getDouble(2), equalTo(4.0));
41          assertThat(dataset.getValue(2), equalTo(7.0));
42          assertThat(dataset.getXBoundaries().getDouble(3), equalTo(6.0));
43          assertThat(dataset.getValue(3), equalTo(-1.0));
44          assertThat(dataset.getXBoundaries().getDouble(4), equalTo(8.0));
45          assertThat(dataset.getValue(4), equalTo(2.0));
46          assertThat(dataset.getXBoundaries().getDouble(5), equalTo(10.0));
47      }
48      
49      @Test
50      public void datasetFrom1() {
51          ListNumber values = new ArrayDouble(5, 3, 7, -1, 2);
52          ListNumber boundaries = new ArrayDouble(0, 2, 4, 6, 8, 10);
53          Cell1DDataset dataset = Cell1DDatasets.datasetFrom(values, boundaries);
54          
55          assertThat(dataset.getXCount(), equalTo(5));
56          assertThat(dataset.getXRange().getMinimum(), equalTo(0.0));
57          assertThat(dataset.getXRange().getMaximum(), equalTo(10.0));
58          assertThat(dataset.getStatistics().getAverage(), equalTo(3.2));
59          assertThat(dataset.getStatistics().getStdDev(), closeTo(2.71293, 0.0001));
60          assertThat(dataset.getStatistics().getRange().getMinimum(), equalTo((Number) (-1.0)));
61          assertThat(dataset.getStatistics().getRange().getMaximum(), equalTo((Number) 7.0));
62          assertThat(dataset.getStatistics().getCount(), equalTo(5));
63          
64          // Check values
65          assertThat(dataset.getXBoundaries().getDouble(0), equalTo(0.0));
66          assertThat(dataset.getValue(0), equalTo(5.0));
67          assertThat(dataset.getXBoundaries().getDouble(1), equalTo(2.0));
68          assertThat(dataset.getValue(1), equalTo(3.0));
69          assertThat(dataset.getXBoundaries().getDouble(2), equalTo(4.0));
70          assertThat(dataset.getValue(2), equalTo(7.0));
71          assertThat(dataset.getXBoundaries().getDouble(3), equalTo(6.0));
72          assertThat(dataset.getValue(3), equalTo(-1.0));
