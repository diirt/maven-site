42      private Cell1DDataset hist;
43      private AreaGraph2DRenderer renderer = new AreaGraph2DRenderer(300, 200);
44  
45      public void setHistogram(Cell1DDataset hist) {
46          this.hist = hist;
47          imagePanel.setImage(new BufferedImage(renderer.getImageWidth(), renderer.getImageHeight(), BufferedImage.TYPE_3BYTE_BGR));
48          pack();
49          redrawHistogram();
50      }
51      
52      private int count = 0;
53      private long totalTime;
54      
55      private void redrawHistogram() {
56          count++;
57          long startTime = System.nanoTime();
58          BufferedImage image = (BufferedImage) imagePanel.getImage();
59          Graphics2D graphics = (Graphics2D) image.getGraphics();
60          renderer.draw(graphics, hist);
61          imagePanel.setImage(image);
62          long endTime = System.nanoTime();
63          totalTime += endTime - startTime;
64          if (count % 100 == 0)
65              System.out.println(count + " " + (totalTime / count));
66      }
