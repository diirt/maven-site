24      @Test
25      public void linearRangeFromFunction2D() {
26          Cell2DDataset dataset = Cell2DDatasets.linearRange(new Cell2DDatasets.Function2D() {
27              @Override
28              public double getValue(double x, double y) {
29                  return x + y;
30              }
31          }, Ranges.range(0, 1), 10, Ranges.range(0, 1), 20);
32  
33          assertThat(dataset.getXCount(), equalTo(10));
34          assertThat(dataset.getXRange().getMinimum(), equalTo(0.0));
35          assertThat(dataset.getXRange().getMaximum(), equalTo(1.0));
36          assertThat(dataset.getYCount(), equalTo(20));
37          assertThat(dataset.getYRange().getMinimum(), equalTo(0.0));
38          assertThat(dataset.getYRange().getMaximum(), equalTo(1.0));
39          assertThat(dataset.getStatistics().getAverage(), closeTo(1.0, 0.0001));
40          assertThat(dataset.getStatistics().getStdDev(), closeTo(0.40697, 0.0001));
41          assertThat(dataset.getStatistics().getCount(), equalTo(200));
42          assertThat(dataset.getStatistics().getRange().getMinimum(), closeTo(0.075, 0.0001));
43          assertThat(dataset.getStatistics().getRange().getMaximum(), closeTo(1.925, 0.0001));
44  
45          // Check values
46          assertThat(dataset.getXBoundaries().getDouble(0), equalTo(0.0));
47          assertThat(dataset.getYBoundaries().getDouble(0), equalTo(0.0));
48          assertThat(dataset.getValue(0, 0), closeTo(0.075, 0.0001));
49          assertThat(dataset.getValue(9, 19), closeTo(1.925, 0.0001));
50          assertThat(dataset.getXBoundaries().getDouble(10), equalTo(1.0));
51          assertThat(dataset.getYBoundaries().getDouble(20), equalTo(1.0));
52      }

65      @Test
66      public void datasetFromListNumber() {
67          ListNumber values = new ArrayDouble(5, 3, 7, -1, 2, 2);
68          ListNumber xBoundaries = new ArrayDouble(-1, 0, 1);
69          ListNumber yBoundaries = new ArrayDouble(0, 1, 2, 3);
70          Cell2DDataset dataset = Cell2DDatasets.datasetFrom(values, xBoundaries, yBoundaries);
71  
72          assertThat(dataset.getXCount(), equalTo(2));
73          assertThat(dataset.getXRange().getMinimum(), equalTo(-1.0));
74          assertThat(dataset.getXRange().getMaximum(), equalTo(1.0));
75          assertThat(dataset.getYCount(), equalTo(3));
76          assertThat(dataset.getYRange().getMinimum(), equalTo(0.0));
77          assertThat(dataset.getYRange().getMaximum(), equalTo(3.0));
78          assertThat(dataset.getStatistics().getAverage(), equalTo(3.0));
79          assertThat(dataset.getStatistics().getStdDev(), closeTo(2.51661, 0.0001));
80          assertThat(dataset.getStatistics().getRange().getMinimum(), equalTo((Number) (-1.0)));
81          assertThat(dataset.getStatistics().getRange().getMaximum(), equalTo((Number) 7.0));
82          assertThat(dataset.getStatistics().getCount(), equalTo(6));
83  
84          // Check values
85          assertThat(dataset.getXBoundaries().getDouble(0), equalTo(-1.0));
86          assertThat(dataset.getXBoundaries().getDouble(1), equalTo(0.0));
87          assertThat(dataset.getXBoundaries().getDouble(2), equalTo(1.0));
88          assertThat(dataset.getYBoundaries().getDouble(0), equalTo(0.0));
89          assertThat(dataset.getYBoundaries().getDouble(1), equalTo(1.0));
90          assertThat(dataset.getYBoundaries().getDouble(2), equalTo(2.0));
91          assertThat(dataset.getYBoundaries().getDouble(3), equalTo(3.0));
92          assertThat(dataset.getValue(0, 0), equalTo(5.0));
93          assertThat(dataset.getValue(1, 0), equalTo(3.0));
94          assertThat(dataset.getValue(0, 1), equalTo(7.0));
95          assertThat(dataset.getValue(1, 1), equalTo(-1.0));
96          assertThat(dataset.getValue(0, 2), equalTo(2.0));
97          assertThat(dataset.getValue(1, 2), equalTo(2.0));
98      }

100     @Test
101     public void datasetFromFunction2D() {
102         Cell2DDataset dataset = Cell2DDatasets.datasetFrom(new Cell2DDatasets.Function2D() {
103             @Override
104             public double getValue(double x, double y) {
105                 return x + y;
106             }
107         }, new ArrayDouble(0, 2, 3, 5), new ArrayDouble(0, 1, 3));
108 
109         assertThat(dataset.getXCount(), equalTo(3));
110         assertThat(dataset.getXRange().getMinimum(), equalTo(0.0));
111         assertThat(dataset.getXRange().getMaximum(), equalTo(5.0));
112         assertThat(dataset.getYCount(), equalTo(2));
113         assertThat(dataset.getYRange().getMinimum(), equalTo(0.0));
114         assertThat(dataset.getYRange().getMaximum(), equalTo(3.0));
115         assertThat(dataset.getStatistics().getAverage(), equalTo(3.75));
116         assertThat(dataset.getStatistics().getStdDev(), closeTo(1.43614, 0.0001));
117         assertThat(dataset.getStatistics().getRange().getMinimum(), equalTo((Number) (1.5)));
118         assertThat(dataset.getStatistics().getRange().getMaximum(), equalTo((Number) 6.0));
119         assertThat(dataset.getStatistics().getCount(), equalTo(6));
120 
121         // Check values
122         assertThat(dataset.getXBoundaries().getDouble(0), equalTo(0.0));
123         assertThat(dataset.getXBoundaries().getDouble(1), equalTo(2.0));
124         assertThat(dataset.getXBoundaries().getDouble(2), equalTo(3.0));
125         assertThat(dataset.getXBoundaries().getDouble(3), equalTo(5.0));
126         assertThat(dataset.getYBoundaries().getDouble(0), equalTo(0.0));
127         assertThat(dataset.getYBoundaries().getDouble(1), equalTo(1.0));
128         assertThat(dataset.getYBoundaries().getDouble(2), equalTo(3.0));
129         assertThat(dataset.getValue(0, 0), equalTo(1.5));
130         assertThat(dataset.getValue(1, 0), equalTo(3.0));
131         assertThat(dataset.getValue(2, 0), equalTo(4.5));
132         assertThat(dataset.getValue(0, 1), equalTo(3.0));
133         assertThat(dataset.getValue(1, 1), equalTo(4.5));
134         assertThat(dataset.getValue(2, 1), equalTo(6.0));
135     }
