59      private void redrawGraph() {
60          BufferedImage newImage = new BufferedImage(getRootPane().getWidth(), getRootPane().getHeight(), BufferedImage.TYPE_3BYTE_BGR);
61          renderer.redrawHistogram(newImage);

63      }

76      public static void showHistogram(final Cell1DDataset hist) {
77          final AreaGraph2DRenderer renderer = new AreaGraph2DRenderer(200, 300);
78          showGraph(new Renderer() {
79  
80              @Override
81              public void redrawHistogram(BufferedImage image) {
82                  renderer.update(new AreaGraph2DRendererUpdate()
83                          .imageHeight(image.getHeight())
84                          .imageWidth(image.getWidth()));
85                  renderer.draw(image.createGraphics(), hist);
86              }
87          });
88      }
