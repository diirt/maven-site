61      public void setHistogram(Cell1DDataset hist) {
62          this.hist = hist;
63          pack();
64          redrawHistogram();
65      }
66      
67      private int count = 0;
68      private long totalTime;
69      
70      private void redrawHistogram() {
71          count++;
72          long startTime = System.nanoTime();
73          BufferedImage image = (BufferedImage) imagePanel.getImage();
74          Graphics2D graphics = (Graphics2D) image.getGraphics();
75          renderer.draw(graphics, hist);
76          imagePanel.setImage(image);
77          long endTime = System.nanoTime();
78          totalTime += endTime - startTime;
79          if (count % 100 == 0)
80              System.out.println(count + " " + (totalTime / count));
81      }
