25      @Test
26      public void xScalingAsCell() throws Exception {
27          GraphBuffer graph = new GraphBuffer(300, 200);
28          graph.setXScaleAsCell(Ranges.range(0, 100), 0, 3, ValueScales.linearScale());
29          assertThat(graph.xValueToPixel(0), equalTo(0));
30          assertThat(graph.xValueToPixel(24.9999), equalTo(0));
31          assertThat(graph.xValueToPixel(25), equalTo(1));
32          assertThat(graph.xValueToPixel(49.9999), equalTo(1));
33          assertThat(graph.xValueToPixel(50), equalTo(2));
34          assertThat(graph.xValueToPixel(74.9999), equalTo(2));
35          assertThat(graph.xValueToPixel(75), equalTo(3));
36          assertThat(graph.xValueToPixel(99.9999), equalTo(3));
37          assertThat(graph.xValueToPixel(100), equalTo(4));
38          assertThat(graph.xPixelLeftToValue(0), equalTo(0.0));
39          assertThat(graph.xPixelCenterToValue(0), equalTo(12.5));
40          assertThat(graph.xPixelRightToValue(0), equalTo(25.0));
41          assertThat(graph.xPixelLeftToValue(1), equalTo(25.0));
42          assertThat(graph.xPixelCenterToValue(1), equalTo(37.5));
43          assertThat(graph.xPixelRightToValue(1), equalTo(50.0));
44          assertThat(graph.xPixelLeftToValue(2), equalTo(50.0));
45          assertThat(graph.xPixelCenterToValue(2), equalTo(62.5));
46          assertThat(graph.xPixelRightToValue(2), equalTo(75.0));
47          assertThat(graph.xPixelLeftToValue(3), equalTo(75.0));
48          assertThat(graph.xPixelCenterToValue(3), equalTo(87.5));
49          assertThat(graph.xPixelRightToValue(3), equalTo(100.0));
50      }

126         graphBuffer.drawBackground(Color.WHITE);
