24          GraphAreaData data = new GraphAreaData();
25          data.setGraphArea(5, 194, 294, 5);
26          data.setGraphPadding(0, 0, 0, 0);
27          data.setGraphBuffer(graphBuffer);
28          data.setLabelMargin(5, 5);
29          data.setRanges(Ranges.range(-5, 5), ValueScales.linearScale(), Ranges.range(-10, 10), ValueScales.linearScale());
30          graphBuffer.drawBackground(Color.WHITE);
31          data.prepareLabels(FontUtil.getLiberationSansRegular(), Color.BLACK);
32          data.prepareGraphArea(true, Color.GRAY);
33          data.drawGraphArea();
34          
35          ImageAssert.compareImages("graphAreaData.linearGraphRange", graphBuffer.getImage());
