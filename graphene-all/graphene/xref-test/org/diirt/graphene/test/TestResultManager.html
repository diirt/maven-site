5   package org.diirt.graphene.test;
6   
7   import java.io.File;
8   import java.io.FilenameFilter;
9   import java.util.LinkedList;
10  import java.util.List;
11  
12  /**
13   *
14   * @author carcassi
15   */
16  public class TestResultManager {
17      private File outputDirectory = new File("src/test/resources/org/epics/graphene");
18  
19      public TestResultManager() {
20          refresh();
21      }
22      
23      
24      public class Result {
25          private File failedImage;
26          private File referenceImage;
27  
28          public Result(File failedImage, File referenceImage) {
29              this.failedImage = failedImage;
30              this.referenceImage = referenceImage;
31          }
32          
33          public void accept() {
34              TestResultManager.this.accept(this);
35          }
36  
37          public File getFailedImage() {
38              return failedImage;
39          }
40  
41          public File getReferenceImage() {
42              return referenceImage;
43          }
44  
45          @Override
46          public String toString() {
47              return referenceImage.getName();
48          }
49          
50      }
51      
52      private List<Result> currentResults;
53      
54      public void refresh() {
55          File[] failedFiles = outputDirectory.listFiles(new FilenameFilter() {
56              
57              @Override
58              public boolean accept(File dir, String name) {
59                  return name.endsWith("failed.png");
60              }
61          });
62          currentResults = new LinkedList<>();
63          for (int i = 0; i < failedFiles.length; i++) {
64              File file = failedFiles[i];
65              File referenceFile = new File(file.getParentFile(), file.getName().substring(0, file.getName().length() - 10) + "png");
66              currentResults.add(new Result(file, referenceFile));
67          }
68      }
69  
70      public List<Result> getCurrentResults() {
71          return currentResults;
72      }
73      
74      public void accept(Result result) {
75          if (result.referenceImage.exists()) {
76              result.referenceImage.delete();
77          }
78          
79          result.failedImage.renameTo(result.referenceImage);
80          currentResults.remove(result);
81      }
82      
83      public static void main(String[] args) {
84          TestResultManager manager = new TestResultManager();
85          manager.refresh();
86          System.out.println(manager.currentResults);
87      }
88  }
