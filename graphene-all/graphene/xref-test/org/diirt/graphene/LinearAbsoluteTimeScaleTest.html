27      @Test
28      public void scaleNormalizedTime1() {
29          TimeScale linearScale = TimeScales.linearAbsoluteScale();
30          assertThat(linearScale.scaleNormalizedTime(0.0, 1.0, 100.0), equalTo(1.0));
31          assertThat(linearScale.scaleNormalizedTime(0.25, 1.0, 100.0), equalTo(25.75));
32          assertThat(linearScale.scaleNormalizedTime(0.5, 1.0, 100.0), equalTo(50.5));
33          assertThat(linearScale.scaleNormalizedTime(1.0, 1.0, 100.0), equalTo(100.0));
34      }
35  
36      @Test
37      public void scaleTimestamp1() {
38          TimeScale linearScale = TimeScales.linearAbsoluteScale();
39          Timestamp start = Timestamp.now();
40          TimeInterval timeInterval = TimeInterval.between(start, start.plus(TimeDuration.ofSeconds(8)));
41          assertThat(linearScale.scaleTimestamp(start, timeInterval, 1.0, 100.0), equalTo(1.0));
42          assertThat(linearScale.scaleTimestamp(start.plus(TimeDuration.ofSeconds(2)), timeInterval, 1.0, 100.0), equalTo(25.75));
43          assertThat(linearScale.scaleTimestamp(start.plus(TimeDuration.ofSeconds(4)), timeInterval, 1.0, 100.0), equalTo(50.5));
44          assertThat(linearScale.scaleTimestamp(start.plus(TimeDuration.ofSeconds(8)), timeInterval, 1.0, 100.0), equalTo(100.0));
45      }

125     public static void assertAxisEquals(TimeInterval timeInterval, ListDouble normalizedValues, List<Timestamp> timestamps, List<String> labels, TimeAxis axis) {
126         assertThat(axis.getTimeInterval(), equalTo(timeInterval));
127         assertThat(axis.getNormalizedValues(), equalTo(normalizedValues));
128         assertThat(axis.getTimestamps(), equalTo(timestamps));
129         assertThat(axis.getTickLabels(), equalTo(labels));
130     }
