33      private Point3DWithLabelDataset defaultDataset() {
34          Random rand = new Random(0);
35          int size = 50;
36          ArrayDouble x = new ArrayDouble(new double[size], false);
37          ArrayDouble y = new ArrayDouble(new double[size], false);
38          ArrayDouble z = new ArrayDouble(new double[size], false);
39          String[] labelSet = new String[]{"First", "Second", "Third", "Fourth", "Fifth"};
40          List<String> labels = new ArrayList<String>(size);
41          for (int i = 0; i < size; i++) {
42              x.setDouble(i, rand.nextGaussian());
43              y.setDouble(i, rand.nextGaussian());
44              z.setDouble(i, 5.0 + rand.nextGaussian());
45              labels.add(labelSet[rand.nextInt(labelSet.length)]);
46          }
47  
48          Point3DWithLabelDataset data = Point3DWithLabelDatasets.build(x, y, z, labels);
49          return data;
50      }

140     @Test
141     public void randomValues() throws Exception {
142 
143         Point3DWithLabelDataset data = randomDataset();
144         BufferedImage image = new BufferedImage(300, 200, BufferedImage.TYPE_3BYTE_BGR);
145         BubbleGraph2DRenderer renderer = new BubbleGraph2DRenderer(300, 200);
146         Graphics2D graphics = (Graphics2D) image.getGraphics();
147         renderer.draw(graphics, data);
148         ImageAssert.compareImages("bubbleGraph2D.randomValues", image);
149 
150         renderer.update(renderer.newUpdate().highlightFocusValue(true).focusPixel(178, 90));
151         graphics = (Graphics2D) image.getGraphics();
152         renderer.draw(graphics, data);
153         ImageAssert.compareImages("bubbleGraph2D.focusPixel1", image);
154         assertThat(renderer.getFocusValueIndex(), equalTo(48));
155 
156         renderer.update(renderer.newUpdate().highlightFocusValue(true).focusPixel(171, 88));
157         graphics = (Graphics2D) image.getGraphics();
158         renderer.draw(graphics, data);
159         ImageAssert.compareImages("bubbleGraph2D.focusPixel2", image);
160         assertThat(renderer.getFocusValueIndex(), equalTo(6));
161     }
