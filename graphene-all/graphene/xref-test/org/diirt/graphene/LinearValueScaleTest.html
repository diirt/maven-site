205     public static void assertAxisEquals(double minValue, double maxValue, double[] tickValues, String[] tickLabels, org.diirt.graphene.ValueAxis axis) {
206         assertEquals(minValue, axis.getMinValue(), 0.000001);
207         assertEquals(maxValue, axis.getMaxValue(), 0.000001);
208         assertArrayEquals(tickValues, axis.getTickValues(), 0.000001);
209         assertArrayEquals(tickLabels, axis.getTickLabels());
210     }
