180     private void assertAxisEquals(double minValue, double maxValue, double[] tickValues, String[] tickLabels, org.diirt.graphene.ValueAxis axis) {
181         assertEquals(minValue, axis.getMinValue(), 0.000001);
182         assertEquals(maxValue, axis.getMaxValue(), 0.000001);
183         assertArrayEquals(tickValues, axis.getTickValues(), 0.000001);
184         assertArrayEquals(tickLabels, axis.getTickLabels());
185     }
