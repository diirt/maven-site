5   package org.diirt.datasource.sample.services.math;
6   
7   import java.util.Arrays;
8   import java.util.Collection;
9   import org.diirt.service.Service;
10  import org.diirt.service.ServiceDescription;
11  import org.diirt.service.ServiceMethod;
12  import org.diirt.service.ServiceMethodDescription;
13  import org.diirt.service.ServiceProvider;
14  import org.diirt.vtype.VNumber;
15  
16  /**
17   * A factory for generation of the Math sample service.
18   *
19   * @author asbarber
20   */
21  public class MathServiceProvider implements ServiceProvider {
22      
23      /**
24       * Creates the service method description for the addition service method.
25       *
26       * @return addition service method description
27       */
28      private static ServiceMethodDescription addMethod() {
29          return new ServiceMethodDescription("add", "Adds two numbers.") {
30  
31              @Override
32              public ServiceMethod createServiceMethod(ServiceDescription serviceDescription) {
33                  return new AddServiceMethod(this, serviceDescription);
34              }
35  
36          }.addArgument("arg1", "First argument", VNumber.class)
37                  .addArgument("arg2", "Second argument", VNumber.class)
38                  .addResult("result", "The sum of arg1 and arg2", VNumber.class);
39      }
40  
41      /**
42       * Creates the service method description for the multiplication service
43       * method.
44       *
45       * @return multiplication service method description
46       */    
47      public static ServiceMethodDescription multiplyMethod() {
48          return new ServiceMethodDescription("multiply", "Multiplies two numbers.") {
49  
50              @Override
51              public ServiceMethod createServiceMethod(ServiceDescription serviceDescription) {
52                  return new MultiplyServiceMethod(this, serviceDescription);
53              }
54  
55          }.addArgument("arg1", "First argument", VNumber.class)
56                  .addArgument("arg2", "Second argument", VNumber.class)
57                  .addResult("result", "The product of arg1 and arg2", VNumber.class);
58      }
59  
60      /**
61       * Creates the the math service.
62       *
63       * @return math service
64       */
65      private static Service createMathService() {
66          return new ServiceDescription("math", "Simple math service for VNumber")
67                  .addServiceMethod(addMethod())
68                  .addServiceMethod(multiplyMethod())
69                  .createService();
70      }
71  
72      @Override
73      public String getName() {
74          // The name of the service provider: this is not the name under which
75          // the services are registered. This is mainly used for logging
76          // purposes.
77          return "mathProvider";
78      }
79  
80      @Override
81      public Collection<Service> createServices() {
82          // Creates all the services that need to be registered
83          return Arrays.asList(createMathService());
84      }
85  
86  }
