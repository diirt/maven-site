5   package org.diirt.datasource.sample.services.math;
6   
7   import java.util.HashMap;
8   import java.util.Map;
9   import org.diirt.service.ServiceDescription;
10  import org.diirt.service.ServiceMethod;
11  import org.diirt.service.ServiceMethodDescription;
12  import org.diirt.vtype.VNumber;
13  import org.diirt.vtype.ValueFactory;
14  
15  /**
16   * An example service method for the addition of {@link VNumber}s.
17   *
18   * @author carcassi
19   */
20  public class AddServiceMethod extends ServiceMethod {
21  
22      /**
23       * Creates a service method for adding {@code VNumber}s together.
24       *
25       * @param serviceMethodDescription the description of the add service
26       * method; can't be null
27       * @param serviceDescription the description of the math service; can't be
28       * null
29       */
30      public AddServiceMethod(ServiceMethodDescription serviceMethodDescription, ServiceDescription serviceDescription) {
31          super(serviceMethodDescription, serviceDescription);
32      }
33  
34      @Override
35      public Map<String, Object> syncExecImpl(Map<String, Object> parameters) {
36          // Method implementation: this is what the service method actually does
37          // Here we are choosing a synchronous implementation, that is the
38          // calculation is done on the thread inside this method call.
39          
40          // Extract data from parameters
41          VNumber arg1 = (VNumber) parameters.get("arg1");
42          VNumber arg2 = (VNumber) parameters.get("arg2");
43          
44          // Check for nulls
45          if (arg1 == null || arg2 == null) {
46              return new HashMap<>();
47          }
48          
49          // Perform calculation
50          VNumber result = ValueFactory.newVDouble(arg1.getValue().doubleValue() + arg2.getValue().doubleValue());
51          
52          // Prepare the result
53          Map<String, Object> resultMap = new HashMap<>();
54          resultMap.put("result", result);
55          return resultMap;
56      }
57  }
