5   package org.diirt.datasource.sample.services.math;
6   
7   import java.util.HashMap;
8   import java.util.Map;
9   import org.diirt.service.ServiceMethod;
10  import org.diirt.service.ServiceRegistry;
11  import org.diirt.vtype.ValueFactory;
12  
13  /**
14   * Locates a service method and executes it synchronously (i.e. waits for the
15   * call to finish).
16   *
17   * @author asbarber
18   */
19  public class SynchonousMethodExecutionExample {
20  
21      /**
22       * Sample execution of Math services.
23       *
24       * @param args ignored
25       */
26      public static void main(String[] args) {
27          System.out.println("Locating service method...");
28          ServiceMethod method = ServiceRegistry.getDefault().findServiceMethod("math/add");
29          System.out.println("Service method found: " + method);
30          
31          System.out.println("Preparing arguments...");
32          Map<String, Object> arguments = new HashMap<>();
33          arguments.put("arg1", ValueFactory.newVDouble(1.0));
34          arguments.put("arg2", ValueFactory.newVDouble(2.0));
35          System.out.println("Arguments: " + arguments);
36          
37          System.out.println("Executing service...");
38          Map<String, Object> result = method.executeSync(arguments);
39          System.out.println("Result: " + result);
40      }
41  
42  }
